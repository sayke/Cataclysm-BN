345int avatar::time_to_read( const item &book, const player &reader, const player *learner ) const
346{
347 const auto &type = book.type->book;
348 const skill_id &skill = type->skill;
349 // The reader's reading speed has an effect only if they're trying to understand the book as they read it
350 // Reading speed is assumed to be how well you learn from books (as opposed to hands-on experience)
351 const bool try_understand = character_funcs::is_fun_to_read( reader, book ) ||
352 reader.get_skill_level( skill ) < type->level;
353 int reading_speed = try_understand ? std::max( reader.read_speed(), read_speed() ) : read_speed();
354 if( learner ) {
355 reading_speed = std::max( reading_speed, learner->read_speed() );
356 }
357
358 int retval = type->time * reading_speed;
359 retval *= std::min( character_funcs::fine_detail_vision_mod( *this ),
360 character_funcs::fine_detail_vision_mod( reader ) );
361
362 const int effective_int = std::min( { get_int(), reader.get_int(), learner ? learner->get_int() : INT_MAX } );
363 if( type->intel > effective_int && !reader.has_trait( trait_PROF_DICEMASTER ) ) {
364 retval += type->time * ( type->intel - effective_int ) * 100;
365 }
366 return retval;
367}

626 if( complex_penalty && !continuous ) {
627 add_msg( m_warning,
628 _( "This book is too complex for %s to easily understand. It will take longer to read." ),
629 complex_player->disp_name() );
630 }
631
632 // push an indentifier of martial art book to the action handling
633 if( it.type->use_methods.count( "MA_MANUAL" ) ) {
634
635 if( get_stamina() < get_stamina_max() / 10 ) {
636 add_msg( m_info, _( "You are too exhausted to train martial arts." ) );
637 return false;
638 }
639 act.str_values.clear();
640 act.str_values.emplace_back( "martial_art" );
641 }
642
643 assign_activity( act );
644
645 // Reinforce any existing morale bonus/penalty, so it doesn't decay
646 // away while you read more.
647 const time_duration decay_start = 1_turns * time_taken / 1000;

658 decay_start, false, it.type );
659 }

742void avatar::do_read( item_location loc )
743{
744 if( !loc ) {
745 activity.set_to_null();
746 return;
747 }
748
749 item &book = *loc;
750 const auto &reading = book.type->book;
751 if( !reading ) {
752 activity.set_to_null();
753 return;
754 }
755 const skill_id &skill = reading->skill;
756
757 if( !has_identified( book.typeId() ) ) {
758 // Note that we've read the book.
759 items_identified.insert( book.typeId() );
760 skim_book_msg( book, *this );
761 activity.set_to_null();
762 return;
763 }
764
765 const bool allow_recipes = get_option<bool>( "ALLOW_LEARNING_BOOK_RECIPES" );
766
767 //learners and their penalties
768 std::vector<std::pair<player *, double>> learners;
769 for( size_t i = 0; i < activity.values.size(); i++ ) {
770 player *n = g->find_npc( character_id( activity.values[i] ) );
771 if( n != nullptr ) {
772 const std::string &s = activity.get_str_value( i, "1" );
773 learners.push_back( { n, strtod( s.c_str(), nullptr ) } );
774 }
775 // Otherwise they must have died/teleported or something
776 }
777 learners.push_back( { this, 1.0 } );
778 //whether to continue reading or not
779 bool continuous = false;
780 // NPCs who learned a little about the skill
781 std::set<std::string> little_learned;
782 std::set<std::string> cant_learn;
783 std::list<std::string> out_of_chapters;
784
785 for( auto &elem : learners ) {
786 player *learner = elem.first;
787
788 int book_fun_for = character_funcs::get_book_fun_for( *learner, book );
789 if( book_fun_for != 0 ) {
790 learner->add_morale( MORALE_BOOK, book_fun_for * 5, book_fun_for * 15, 1_hours, 30_minutes, true,
791 book.type );
792 }
793
794 book.mark_chapter_as_read( *learner );
795
796 const auto available_recipes = book.get_available_recipes( *learner );
797 std::vector<const recipe *> learnable_recipes;
798 for( const std::pair<const recipe *, int> &p : available_recipes ) {
799 if( allow_recipes && !learner->knows_recipe( p.first ) ) {
800 learnable_recipes.push_back( p.first );
801 learner->learn_recipe( p.first );
802 if( learner->is_player() ) {
803 add_msg( m_info, _( "You memorize a recipe for %s." ), p.first->result_name() );
804 }
805 }
806 }
807
808 if( skill && learner->get_skill_level( skill ) < reading->level &&
809 learner->get_skill_level_object( skill ).can_train() ) {
810 SkillLevel &skill_level = learner->get_skill_level_object( skill );
811 const int originalSkillLevel = skill_level.level();
812
813 // Calculate experience gained
814 /** @EFFECT_INT increases reading comprehension */
815 // Enhanced Memory Banks modestly boosts experience
816 int min_ex = std::max( 1, reading->time / 10 + learner->get_int() / 4 );
817 int max_ex = reading->time / 5 + learner->get_int() / 2 - originalSkillLevel;
818 if( has_active_bionic( bio_memory ) ) {
819 min_ex += 2;
820 }
821
822 min_ex = adjust_for_focus( min_ex );
823 max_ex = adjust_for_focus( max_ex );
824
825 if( max_ex < 2 ) {
826 max_ex = 2;
827 }
828 if( max_ex > 10 ) {
829 max_ex = 10;
830 }
831 if( max_ex < min_ex ) {
832 max_ex = min_ex;
833 }
834
835 min_ex *= ( originalSkillLevel + 1 ) * elem.second;
836 min_ex = std::max( min_ex, 1 );
837 max_ex *= ( originalSkillLevel + 1 ) * elem.second;
838 max_ex = std::max( min_ex, max_ex );
839
840 skill_level.readBook( min_ex, max_ex, reading->level );
841
842 std::string skill_name = skill.obj().name();
843
844 if( skill_level != originalSkillLevel ) {
845 g->events().send<event_type::gains_skill_level>(
846 learner->getID(), skill, skill_level.level() );
847 if( learner->is_player() ) {
848 add_msg( m_good, _( "You increase %s to level %d." ), skill.obj().name(),
849 originalSkillLevel + 1 );
850 } else {
851 add_msg( m_good, _( "%s increases their %s level." ), learner->disp_name(), skill_name );
852 }
853 } else {
854 //skill_level == originalSkillLevel
855 if( activity.index == learner->getID().get_value() ) {
856 continuous = true;
857 }
858 if( learner->is_player() ) {
859 add_msg( m_info, _( "You learn a little about %s! (%d%%)" ), skill_name, skill_level.exercise() );
860 } else {
861 little_learned.insert( learner->disp_name() );
862 }
863 }
864
865 if( learnable_recipes.empty() && ( skill_level == reading->level || !skill_level.can_train() ) ) {
866 if( learner->is_player() ) {
867 add_msg( m_info, _( "You can no longer learn from %s." ), book.type_name() );
868 } else {
869 cant_learn.insert( learner->disp_name() );
870 }
871 }
872 } else if( learnable_recipes.empty() && skill ) {
873 if( learner->is_player() ) {
874 add_msg( m_info, _( "You can no longer learn from %s." ), book.type_name() );
875 } else {
876 cant_learn.insert( learner->disp_name() );
877 }
878 }
879 }
880 //end for all learners
881
882 if( little_learned.size() == 1 ) {
883 add_msg( m_info, _( "%s learns a little about %s!" ), little_learned.begin()->c_str(),
884 skill.obj().name() );
885 } else if( !little_learned.empty() ) {
886 const std::string little_learned_msg = enumerate_as_string( little_learned );
887 add_msg( m_info, _( "%s learn a little about %s!" ), little_learned_msg, skill.obj().name() );
888 }
889
890 if( !cant_learn.empty() ) {
891 const std::string names = enumerate_as_string( cant_learn );
892 add_msg( m_info, _( "%s can no longer learn from %s." ), names, book.type_name() );
893 }
894 if( !out_of_chapters.empty() ) {
895 const std::string names = enumerate_as_string( out_of_chapters );
896 add_msg( m_info, _( "Rereading the %s isn't as much fun for %s." ),
897 book.type_name(), names );
898 if( out_of_chapters.front() == disp_name() && one_in( 6 ) ) {
899 add_msg( m_info, _( "Maybe you should find something new to read…" ) );
900 }
901 }
902
903 // NPCs can't learn martial arts from manuals (yet).
904 auto m = book.type->use_methods.find( "MA_MANUAL" );
905 if( m != book.type->use_methods.end() ) {
906 const matype_id style_to_learn = martial_art_learned_from( *book.type );
907 skill_id skill_used = style_to_learn->primary_skill;
908 int difficulty = std::max( 1, style_to_learn->learn_difficulty );
909 difficulty = std::max( 1, 20 + difficulty * 2 - get_skill_level( skill_used ) * 2 );
910 add_msg( m_debug, _( "Chance to learn one in: %d" ), difficulty );
911
912 if( one_in( difficulty ) ) {
913 m->second.call( *this, book, false, pos() );
914 continuous = false;
915 } else {
916 if( activity.index == getID().get_value() ) {
917 continuous = true;
918 switch( rng( 1, 5 ) ) {
919 case 1:
920 add_msg( m_info,
921 _( "You train the moves according to the book, but can't get a grasp of the style, so you start from the beginning." ) );
922 break;
923 case 2:
924 add_msg( m_info,
925 _( "This martial art is not easy to grasp. You start training the moves from the beginning." ) );
926 break;
927 case 3:
928 add_msg( m_info,
929 _( "You decide to read the manual and train even more. In martial arts, patience leads to mastery." ) );
930 break;
931 case 4:
932 case 5:
933 add_msg( m_info, _( "You try again. This training will finally pay off." ) );
934 break;
935 }
936 } else {
937 add_msg( m_info, _( "You train for a while." ) );
938 }
939 }
940 }
941
942 if( continuous ) {
943 activity.set_to_null();
944 read( loc, true );
945 if( activity ) {
946 return;
947 }
948 }
949
950 activity.set_to_null();
951}

1199bool avatar::wield( item &target )
1200{
1201 if( is_wielding( target ) ) {
1202 return true;
1203 }
1204
1205 if( !can_wield( target ).success() ) {
1206 return false;
1207 }
1208
1209 if( !unwield() ) {
1210 return false;
1211 }
1212 clear_npc_ai_info_cache( npc_ai_info::ideal_weapon_value );
1213 if( target.is_null() ) {
1214 return true;
1215 }
1216
1217 // Query whether to draw an item from a holster when attempting to wield the holster
1218 if( target.get_use( "holster" ) && !target.contents.empty() ) {
1219 //~ %1$s: weapon name, %2$s: holster name
1220 if( query_yn( pgettext( "holster", "Draw %1$s from %2$s?" ), target.get_contained().tname(),
1221 target.tname() ) ) {
1222 invoke_item( &target );
1223 return false;
1224 }
1225 }
1226
1227 // Wielding from inventory is relatively slow and does not improve with increasing weapon skill.
1228 // Worn items (including guns with shoulder straps) are faster but still slower
1229 // than a skilled player with a holster.
1230 // There is an additional penalty when wielding items from the inventory whilst currently grabbed.
1231
1232 bool worn = is_worn( target );
1233 int mv = item_handling_cost( target, true,
1234 worn ? INVENTORY_HANDLING_PENALTY / 2 : INVENTORY_HANDLING_PENALTY );
1235
1236 add_msg( m_debug, "wielding took %d moves", mv );
1237 moves -= mv;
1238
1239 if( has_item( target ) ) {
1240 set_primary_weapon( i_rem( &target ) );
1241 } else {
1242 set_primary_weapon( target );
1243 }
1244
1245 last_item = primary_weapon().typeId();
1246 recoil = MAX_RECOIL;
1247
1248 primary_weapon().on_wield( *this, mv );
1249
1250 inv.update_invlet( primary_weapon() );
1251 inv.update_cache_with_item( primary_weapon() );
1252
1253 return true;
1254}
