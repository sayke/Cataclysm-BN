29static double pow10( unsigned int n )
30{
31 double ret = 1;
32 double tmp = 10;
33 while( n ) {
34 if( n & 1 ) {
35 ret *= tmp;
36 }
37 tmp *= tmp;
38 n >>= 1;
39 }
40 return ret;
41}

51int modulo( int v, int m )
52{
53 // C++11: negative v and positive m result in negative v%m (or 0),
54 // but this is supposed to be mathematical modulo: 0 <= v%m < m,
55 const int r = v % m;
56 // Adding m in that (and only that) case.
57 return r >= 0 ? r : r + ( m * ( 1 - r / m ) );
58}

74double logarithmic_range( int min, int max, int pos )
75{
76 const double LOGI_CUTOFF = 4;
77 const double LOGI_MIN = logarithmic( -LOGI_CUTOFF );
78 const double LOGI_MAX = logarithmic( +LOGI_CUTOFF );
79 const double LOGI_RANGE = LOGI_MAX - LOGI_MIN;
80
81 if( min >= max ) {
82 debugmsg( "Invalid interval (%d, %d).", min, max );
83 return 0.0;
84 }
85
86 // Anything beyond (min,max) gets clamped.
87 if( pos <= min ) {
88 return 1.0;
89 } else if( pos >= max ) {
90 return 0.0;
91 }
92
93 // Normalize the pos to [0,1]
94 double range = max - min;
95 double unit_pos = ( pos - min ) / range;
96
97 // Scale and flip it to [+LOGI_CUTOFF,-LOGI_CUTOFF]
98 double scaled_pos = LOGI_CUTOFF - 2 * LOGI_CUTOFF * unit_pos;
99
100 // Get the raw logistic value.
101 double raw_logistic = logarithmic( scaled_pos );
102
103 // Scale the output to [0,1]
104 return ( raw_logistic - LOGI_MIN ) / LOGI_RANGE;
105}

128double clamp_to_width( double value, int width, int &scale, bool *out_truncated )
129{
130 if( out_truncated != nullptr ) {
131 *out_truncated = false;
132 }
133 if( value >= std::pow( 10.0, width ) ) {
134 // above the maximum number we can fit in the width without decimal
135 // show the biggest number we can without decimal
136 // flag as truncated
137 value = std::pow( 10.0, width ) - 1.0;
138 scale = 0;
139 if( out_truncated != nullptr ) {
140 *out_truncated = true;
141 }
142 } else if( scale > 0 ) {
143 for( int s = 1; s <= scale; s++ ) {
144 // 1 decimal separator + "s"
145 int scale_width = 1 + s;
146 if( width > scale_width && value >= std::pow( 10.0, width - scale_width ) ) {
147 // above the maximum number we can fit in the width with "s" decimals
148 // show this number with one less decimal than "s"
149 scale = s - 1;
150 break;
151 }
152 }
153 }
154 return value;
155}

157float multi_lerp( const std::vector<std::pair<float, float>> &points, float x )
158{
159 size_t i = 0;
160 while( i < points.size() && points[i].first <= x ) {
161 i++;
162 }
163
164 if( i == 0 ) {
165 return points.front().second;
166 } else if( i >= points.size() ) {
167 return points.back().second;
168 }
169
170 // How far are we along the way from last threshold to current one
171 const float t = ( x - points[i - 1].first ) /
172 ( points[i].first - points[i - 1].first );
173
174 // Linear interpolation of values at relevant thresholds
175 return ( t * points[i].second ) + ( ( 1 - t ) * points[i - 1].second );
176}

493std::istream &safe_getline( std::istream &ins, std::string &str )
494{
495 str.clear();
496 std::istream::sentry se( ins, true );
497 std::streambuf *sb = ins.rdbuf();
498
499 while( true ) {
500 int c = sb->sbumpc();
501 switch( c ) {
502 case '\n':
503 return ins;
504 case '\r':
505 if( sb->sgetc() == '\n' ) {
506 sb->sbumpc();
507 }
508 return ins;
509 case EOF:
510 if( str.empty() ) {
511 ins.setstate( std::ios::eofbit );
512 }
513 return ins;
514 default:
515 str += static_cast<char>( c );
516 }
517 }

646 if( utf8_width( transformation ) != 1 ) {
647 debugmsg( "target character isn't narrow" );

649 // A 2-width wide character in the original string should be replace by two narrow characters
650 w_str.replace( i, 1, utf8_to_wstr( std::string( utf8_width( this_char ), transformation[0] ) ) );
651 }
652 }
653 std::string result = wstr_to_utf8( w_str );
654 if( utf8_width( str ) != utf8_width( result ) ) {
655 debugmsg( "utf8_width differ between original string and obscured string" );
656 }

704holiday get_holiday_from_time( std::time_t time, bool force_refresh )
705{
706 static holiday cached_holiday = holiday::none;
707 static bool is_cached = false;
708
709 if( force_refresh ) {
710 is_cached = false;
711 }
712 if( is_cached ) {
713 return cached_holiday;
714 }
715
716 is_cached = true;
717
718 bool success = false;
719
720 std::tm local_time;
721 std::time_t current_time = time == 0 ? std::time( nullptr ) : time;
722
723 /* necessary to pass LGTM, as threadsafe version of localtime differs by platform */
724#if defined(_WIN32)
725
726 errno_t err = localtime_s( &local_time, &current_time );
727 if( err == 0 ) {
728 success = true;
729 }
730
731#else
732
733 success = !!localtime_r( &current_time, &local_time );
734
735#endif
736
737 if( success ) {
738
739 const int month = local_time.tm_mon + 1;
740 const int day = local_time.tm_mday;
741 const int wday = local_time.tm_wday;
742 const int year = local_time.tm_year + 1900;
743
744 /* check date against holidays */
745 if( month == 1 && day == 1 ) {
746 cached_holiday = holiday::new_year;
747 return cached_holiday;
748 }
749 // only run easter date calculation if currently March or April
750 else if( ( month == 3 || month == 4 ) && is_easter( day, month, year ) ) {
751 cached_holiday = holiday::easter;
752 return cached_holiday;
753 } else if( month == 7 && day == 4 ) {
754 cached_holiday = holiday::independence_day;
755 return cached_holiday;
756 }
757 // 13 days seems appropriate for Halloween
758 else if( month == 10 && day >= 19 ) {
759 cached_holiday = holiday::halloween;
760 return cached_holiday;
761 } else if( month == 11 && ( day >= 22 && day <= 28 ) && wday == 4 ) {
762 cached_holiday = holiday::thanksgiving;
763 return cached_holiday;
764 }
765 // For the 12 days of Christmas, my true love gave to me...
766 else if( month == 12 && ( day >= 14 && day <= 25 ) ) {
767 cached_holiday = holiday::christmas;
768 return cached_holiday;
