188int itype::invoke( player &p, item &it, const tripoint &pos, const std::string &iuse_name ) const
189{
190 const use_function *use = get_use( iuse_name );
191 if( use == nullptr ) {
192 debugmsg( "Tried to invoke %s on a %s, which doesn't have this use_function",
193 iuse_name, nname( 1 ) );
194 return 0;
195 }
196
197 const auto ret = use->can_call( p, it, false, pos );
198
199 if( !ret.success() ) {
200 p.add_msg_if_player( m_info, ret.str() );
201 return 0;
202 }
203 // used for grenades and such, to increase kill count
204 // invoke is called a first time with transform, when the explosive item is activated
205 // then a second time with draw explosion
206 // the player responsible of the explosion is the one that activated the object
207 if( iuse_name == "transform" ) {
208 it.activated_by = p.get_safe_reference();
209 }
210
211 return use->call( p, it, false, pos );
212}
