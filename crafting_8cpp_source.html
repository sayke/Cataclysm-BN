132float lighting_crafting_speed_multiplier( const Character &who, const recipe &rec )
133{
134 if( character_funcs::can_see_fine_details( who ) ) {
135 return 1.0f;
136 }
137
138 const SkillLevelMap &char_skills = who.get_all_skills();
139 int skill_bonus = char_skills.exceeds_recipe_requirements( rec );
140
141 // This value whould be within [0,1]
142 const float darkness =
143 (
144 character_funcs::fine_detail_vision_mod( who ) -
145 character_funcs::FINE_VISION_THRESHOLD
146 ) / 7.0f;
147
148 if( rec.has_flag( flag_BLIND_EASY ) ) {
149 // 100% speed in well lit area at skill+0
150 // 25% speed in pitch black at skill+0
151 // skill+2 removes speed penalty
152 return 1.0f - darkness * 0.75f * std::max( 0, 2 - skill_bonus ) / 2.0f;
153 } else if( rec.has_flag( flag_BLIND_HARD ) && skill_bonus >= 2 ) {
154 // 100% speed in well lit area at skill+2
155 // 25% speed in pitch black at skill+2
156 // skill+8 removes speed penalty
157 return 1.0f - darkness * 0.75f * std::max( 0, 8 - skill_bonus ) / 6.0f;
158 } else {
159 // Needs proper vision or the character is not skilled enough
160 return 0.0f;
161 }
162}

164float morale_crafting_speed_multiplier( const Character &who, const recipe &rec )
165{
166 int morale = who.get_morale_level();
167 if( morale >= 0 ) {
168 // No bonus for being happy yet
169 return 1.0f;
170 }
171
172 // Harder jobs are more frustrating, even when skilled
173 // For each skill where skill=difficulty, multiply effective morale by 200%
174 float morale_mult = std::max( 1.0f, 2.0f * rec.difficulty / std::max( 1,
175 who.get_skill_level( rec.skill_used ) ) );
176 for( const std::pair<const skill_id, int> &pr : rec.required_skills ) {
177 morale_mult *= std::max( 1.0f, 2.0f * pr.second / std::max( 1, who.get_skill_level( pr.first ) ) );
178 }
179
180 // Halve speed at -50 effective morale, quarter at -150
181 float morale_effect = 1.0f + ( morale_mult * morale ) / -50.0f;
182
183 return 1.0f / morale_effect;
184}

186template<typename T>
187static float lerped_multiplier( const T &value, const T &low, const T &high )
188{
189 // No effect if less than allowed value
190 if( value < low ) {
191 return 1.0f;
192 }
193 // Bottom out at 25% speed
194 if( value > high ) {
195 return 0.25f;
196 }
197 // Linear interpolation between high and low
198 // y = y0 + ( x - x0 ) * ( y1 - y0 ) / ( x1 - x0 )
199 return 1.0f + ( value - low ) * ( 0.25f - 1.0f ) / ( high - low );
200}

203{
204 float multiplier = 0.0f;

206 units::volume allowed_volume;

209 const units::volume &craft_volume = craft.volume();
210
211 // The whole block below is so ugly because all the benches have different structs with same content

216 multiplier = f.workbench->multiplier;
217 allowed_mass = f.workbench->allowed_mass;
218 allowed_volume = f.workbench->allowed_volume;
219 }
220 break;
221 case bench_type::ground: {

226 allowed_volume = f.workbench->allowed_volume;

231 // Furniture workbench

234 allowed_mass = f.workbench->allowed_mass;
235 allowed_volume = f.workbench->allowed_volume;
236 } else {
237 return 0.0f;
238 }
239 break;
240 case bench_type::vehicle:
241 if( const std::optional<vpart_reference> vp = here.veh_at(
242 bench.position ).part_with_feature( "WORKBENCH", true ) ) {
243 // Vehicle workbench
244 const vpart_info &vp_info = vp->part().info();
245 if( const std::optional<vpslot_workbench> &wb_info = vp_info.get_workbench_info() ) {
246 multiplier = wb_info->multiplier;
247 allowed_mass = wb_info->allowed_mass;
248 allowed_volume = wb_info->allowed_volume;

250 debugmsg( "part '%S' with WORKBENCH flag has no workbench info", vp->part().name() );
251 return 0.0f;
252 }
253 }

257 return 0.0f;
258 }

261 multiplier *= lerped_multiplier( craft_mass, allowed_mass, 1000_kilogram );
262 multiplier *= lerped_multiplier( craft_volume, allowed_volume, 1000_liter );
263
264 return multiplier;

268{

283float crafting_speed_multiplier( const Character &who, const item &craft,
284 const bench_location &bench )
285{
286 if( !craft.is_craft() ) {
287 debugmsg( "Can't calculate crafting speed multiplier of non-craft '%s'", craft.tname() );
288 return 1.0f;
289 }
290
291 const recipe &rec = craft.get_making();
292
293 const float light_multi = lighting_crafting_speed_multiplier( who, rec );
294 const float bench_multi = workbench_crafting_speed_multiplier( craft, bench );
295 const float morale_multi = morale_crafting_speed_multiplier( who, rec );
296
297 const float total_multi = light_multi * bench_multi * morale_multi;
298
299 if( light_multi <= 0.0f ) {
300 who.add_msg_if_player( m_bad, _( "You can no longer see well enough to keep crafting." ) );
301 return 0.0f;
302 }
303 if( bench_multi <= 0.1f || ( bench_multi <= 0.33f && total_multi <= 0.2f ) ) {
304 who.add_msg_if_player( m_bad, _( "The %s is too large and/or heavy to work on. You may want to"
305 " use a workbench or a smaller batch size" ), craft.tname() );
306 return 0.0f;
307 }
308 if( morale_multi <= 0.2f || ( morale_multi <= 0.33f && total_multi <= 0.2f ) ) {
309 who.add_msg_if_player( m_bad, _( "Your morale is too low to continue crafting." ) );
310 return 0.0f;
311 }
312
313 // If we're working below 20% speed, just give up
314 if( total_multi <= 0.2f ) {
315 who.add_msg_if_player( m_bad, _( "You are too frustrated to continue and just give up." ) );
316 return 0.0f;
317 }
318
319 if( calendar::once_every( 1_hours ) && total_multi < 0.75f ) {
320 if( light_multi <= 0.5f ) {
321 who.add_msg_if_player( m_bad, _( "You can't see well and are working slowly." ) );
322 }
323 if( bench_multi <= 0.5f ) {
324 who.add_msg_if_player( m_bad,
325 _( "The %s is to large and/or heavy to work on comfortably. You are"
326 " working slowly." ), craft.tname() );
327 }
328 if( morale_multi <= 0.5f ) {
329 who.add_msg_if_player( m_bad, _( "You can't focus and are working slowly." ) );
330 }
331 }
332
333 return total_multi;

414bool player::check_eligible_containers_for_crafting( const recipe &rec, int batch_size ) const
415{
416 std::vector<const item *> conts = get_eligible_containers_for_crafting();
417 const std::vector<item> res = rec.create_results( batch_size );
418 const std::vector<item> bps = rec.create_byproducts( batch_size );
419 std::vector<item> all;
420 all.reserve( res.size() + bps.size() );
421 all.insert( all.end(), res.begin(), res.end() );
422 all.insert( all.end(), bps.begin(), bps.end() );
423
424 map &here = get_map();
425 for( const item &prod : all ) {
426 if( !prod.made_of( LIQUID ) ) {
427 continue;
428 }
429
430 // we go through half-filled containers first, then go through empty containers if we need
431 std::sort( conts.begin(), conts.end(), item_ptr_compare_by_charges );
432
433 int charges_to_store = prod.charges;
434 for( const item *cont : conts ) {
435 if( charges_to_store <= 0 ) {
436 break;
437 }
438
439 if( !cont->is_container_empty() ) {
440 if( cont->contents.front().typeId() == prod.typeId() ) {
441 charges_to_store -= cont->get_remaining_capacity_for_liquid( cont->contents.front(), true );
442 }
443 } else {
444 charges_to_store -= cont->get_remaining_capacity_for_liquid( prod, true );
445 }
446 }
447
448 // also check if we're currently in a vehicle that has the necessary storage
449 if( charges_to_store > 0 ) {
450 if( optional_vpart_position vp = here.veh_at( pos() ) ) {
451 const itype_id &ftype = prod.typeId();
452 int fuel_cap = vp->vehicle().fuel_capacity( ftype );
453 int fuel_amnt = vp->vehicle().fuel_left( ftype );
454
455 if( fuel_cap >= 0 ) {
456 int fuel_space_left = fuel_cap - fuel_amnt;
457 charges_to_store -= fuel_space_left;
458 }
459 }
460 }
461
462 if( charges_to_store > 0 ) {
463 if( !query_yn(
464 _( "You don't have anything in which to store %s and may have to pour it out or consume it as soon as it is prepared! Proceed?" ),
465 prod.tname() ) ) {
466 return false;
467 }
468 }
469 }
470
471 return true;
472}

629static void set_components( std::list<item> &components, const std::list<item> &used,
630 const int batch_size, const size_t offset )
631{
632 if( batch_size <= 1 ) {
633 components.insert( components.begin(), used.begin(), used.end() );
634 return;
635 }
636 // This count does *not* include items counted by charges!
637 size_t non_charges_counter = 0;

642 // which has a charges value that can be evenly divided by batch_size.
643 components.back().charges = tmp.charges / batch_size;
644 } else {
645 if( ( non_charges_counter + offset ) % batch_size == 0 ) {
646 components.push_back( tmp );
647 }
648 non_charges_counter++;
649 }

775{
776 if( !craft.is_craft() ) {
777 debugmsg( "craft_skill_check() called on non-craft '%s.' Aborting.", craft.tname() );
778 return;
779 }
780
781 const recipe &making = craft.get_making();
782 const int batch_size = craft.charges;
783
784 std::vector<npc *> helpers = character_funcs::get_crafting_helpers( *this );
785
786 if( making.skill_used ) {
787 // Normalize experience gain to crafting time, giving a bonus for longer crafting

789 // This is called after every 5% crafting progress, so divide by 20
790 // TODO: Don't multiply, instead divide the crafting time into more "learn bits"
791 const int base_practice = roll_remainder( ( making.difficulty * 15 + 10 ) * batch_mult /

798 //If the NPC can understand what you are doing, they gain more exp
799 if( helper->get_skill_level( making.skill_used ) >= making.difficulty ) {

801 skill_cap );

803 add_msg( m_info, _( "%s assists with crafting…" ), helper->name );
804 }
805 if( batch_size == 1 && one_in( 3 ) ) {
806 add_msg( m_info, _( "%s could assist you with a batch…" ), helper->name );
807 }
808 // NPCs around you understand the skill used better
809 } else {
810 helper->practice( making.skill_used, roll_remainder( base_practice / 10.0 ),
811 skill_cap );
812 if( one_in( 3 ) ) {
813 add_msg( m_info, _( "%s watches you craft…" ), helper->name );
814 }
815 }
816 }

820double player::crafting_success_roll( const recipe &making ) const
821{
822 int secondary_dice = 0;
823 int secondary_difficulty = 0;
824 for( const auto &pr : making.required_skills ) {
825 secondary_dice += get_skill_level( pr.first );
826 secondary_difficulty += pr.second;
827 }
828
829 // # of dice is 75% primary skill, 25% secondary (unless secondary is null)
830 int skill_dice;
831 if( secondary_difficulty > 0 ) {
832 skill_dice = get_skill_level( making.skill_used ) * 3 + secondary_dice;
833 } else {
834 skill_dice = get_skill_level( making.skill_used ) * 4;
835 }
836
837 for( const npc *np : character_funcs::get_crafting_helpers( *this ) ) {
838 if( np->get_skill_level( making.skill_used ) >=
839 get_skill_level( making.skill_used ) ) {
840 // NPC assistance is worth half a skill level
841 skill_dice += 2;
842 add_msg_if_player( m_info, _( "%s helps with crafting…" ), np->name );
843 break;
844 }
845 }
846
847 // farsightedness can impose a penalty on electronics and tailoring success
848 // it's equivalent to a 2-rank electronics penalty, 1-rank tailoring
849 if( has_trait( trait_HYPEROPIC ) && !worn_with_flag( flag_FIX_FARSIGHT ) &&
850 !has_effect( effect_contacts ) ) {
851 int main_rank_penalty = 0;
852 if( making.skill_used == skill_electronics ) {
853 main_rank_penalty = 2;
854 } else if( making.skill_used == skill_tailor ) {
855 main_rank_penalty = 1;
856 }
857 skill_dice -= main_rank_penalty * 4;
858 }
859
860 // It's tough to craft with paws. Fortunately it's just a matter of grip and fine-motor,
861 // not inability to see what you're doing
862 for( const trait_id &mut : get_mutations() ) {
863 for( const std::pair<const skill_id, int> &skib : mut->craft_skill_bonus ) {
864 if( making.skill_used == skib.first ) {
865 skill_dice += skib.second;
866 }
867 }
868 }
869
870 // Sides on dice is 16 plus your current intelligence
871 ///\EFFECT_INT increases crafting success chance
872 const int skill_sides = 16 + int_cur;
873
874 int diff_dice;
875 if( secondary_difficulty > 0 ) {
876 diff_dice = making.difficulty * 3 + secondary_difficulty;
877 } else {
878 // Since skill level is * 4 also
879 diff_dice = making.difficulty * 4;
880 }
881
882 const int diff_sides = 24; // 16 + 8 (default intelligence)
883
884 const double skill_roll = dice( skill_dice, skill_sides );
885 const double diff_roll = dice( diff_dice, diff_sides );
886
887 if( diff_roll == 0 ) {
888 // Automatic success
889 return 2;
890 }
891
892 return skill_roll / diff_roll;
893}

930bool item::handle_craft_failure( player &crafter )
931{
932 if( !is_craft() ) {
933 debugmsg( "handle_craft_failure() called on non-craft '%s.' Aborting.", tname() );
934 return false;
935 }
936
937 const double success_roll = crafter.crafting_success_roll( get_making() );
938 const int starting_components = this->components.size();
939 // Destroy at most 75% of the components, always a chance of losing 1 though
940 const size_t max_destroyed = std::max<size_t>( 1, components.size() * 3 / 4 );
941 for( size_t i = 0; i < max_destroyed; i++ ) {
942 // This shouldn't happen
943 if( components.empty() ) {
944 break;
945 }
946 // If we roll success, skip destroying a component
947 if( x_in_y( success_roll, 1.0 ) ) {
948 continue;
949 }
950 destroy_random_component( *this, crafter );
951 }
952 if( starting_components > 0 && this->components.empty() ) {
953 // The craft had components and all of them were destroyed.
954 return true;
955 }
956
957 // Minimum 25% progress lost, average 35%. Falls off exponentially
958 // Loss is scaled by the success roll
959 const double percent_progress_loss = rng_exponential( 0.25, 0.35 ) *
960 ( 1.0 - std::min( success_roll, 1.0 ) );
961 const int progess_loss = item_counter * percent_progress_loss;
962 crafter.add_msg_player_or_npc( _( "You mess up and lose %d%% progress." ),
963 _( "<npcname> messes up and loses %d%% progress." ), progess_loss / 100000 );
964 item_counter = clamp( item_counter - progess_loss, 0, 10000000 );
965
966 set_next_failure_point( crafter );
967
968 // Check if we can consume a new component and continue
969 if( !crafter.can_continue_craft( *this ) ) {
970 crafter.cancel_activity();
971 }
972 return false;
973}

1019void complete_craft( player &p, item &craft, const bench_location & )
1020{
1021 if( !craft.is_craft() ) {
1022 debugmsg( "complete_craft() called on non-craft '%s.' Aborting.", craft.tname() );
1023 return;
1024 }
1025
1026 const recipe &making = craft.get_making();
1027 const int batch_size = craft.charges;
1028 std::list<item> &used = craft.components;
1029 const double relative_rot = craft.get_relative_rot();
1030 const bool ignore_component = making.has_flag( flag_NUTRIENT_OVERRIDE );
1031
1032 // Set up the new item, and assign an inventory letter if available
1033 std::vector<item> newits = making.create_results( batch_size );
1034
1035 const bool should_heat = making.hot_result();
1036 const bool is_dehydrated = making.dehydrate_result();
1037
1038 bool first = true;
1039 size_t newit_counter = 0;
1040 for( item &newit : newits ) {
1041
1042 // Points to newit unless newit is a non-empty container, then it points to newit's contents.
1043 // Necessary for things like canning soup; sometimes we want to operate on the soup, not the can.
1044 item &food_contained = ( newit.is_container() && !newit.contents.empty() ) ?
1045 newit.contents.back() : newit;
1046
1047 // messages, learning of recipe, food spoilage calculation only once
1048 if( first ) {
1049 first = false;
1050 // TODO: reconsider recipe memorization
1051 if( p.knows_recipe( &making ) ) {
1052 add_msg( _( "You craft %s from memory." ), making.result_name() );
1053 } else {
1054 add_msg( _( "You craft %s using a book as a reference." ), making.result_name() );
1055 // If we made it, but we don't know it,
1056 // we're making it from a book and have a chance to learn it.
1057 // Base expected time to learn is 1000*(difficulty^4)/skill/int moves.
1058 // This means time to learn is greatly decreased with higher skill level,
1059 // but also keeps going up as difficulty goes up.
1060 // Worst case is lvl 10, which will typically take
1061 // 10^4/10 (1,000) minutes, or about 16 hours of crafting it to learn.
1062 int difficulty = p.has_recipe( &making, p.crafting_inventory(),
1063 character_funcs::get_crafting_helpers( p ) );
1064 ///\EFFECT_INT increases chance to learn recipe when crafting from a book
1065 const double learning_speed =
1066 std::max( p.get_skill_level( making.skill_used ), 1 ) *
1067 std::max( p.get_int(), 1 );
1068 const double time_to_learn = 1000 * 8 * std::pow( difficulty, 4 ) / learning_speed;
1069 if( x_in_y( making.time, time_to_learn ) ) {
1070 p.learn_recipe( &making );
1071 add_msg( m_good, _( "You memorized the recipe for %s!" ),
1072 making.result_name() );
1073 }
1074 }
1075 }
1076
1077 food_contained.inherit_flags( used, making );
1078
1079 for( const std::string &flag : making.flags_to_delete ) {
1080 food_contained.unset_flag( flag );
1081 }
1082
1083 // Don't store components for things that ignores components (e.g wow 'conjured bread')
1084 if( ignore_component ) {
1085 food_contained.set_flag( flag_NUTRIENT_OVERRIDE );
1086 } else if( recipe_dictionary::get_uncraft( making.result() ) &&
1087 !food_contained.count_by_charges() &&
1088 making.is_reversible() ) {
1089 // Don't store components for things made by charges,
1090 // Don't store components for things that can't be uncrafted.
1091
1092 // Setting this for items counted by charges gives only problems:
1093 // those items are automatically merged everywhere (map/vehicle/inventory),
1094 // which would either lose this information or merge it somehow.
1095 set_components( food_contained.components, used, batch_size, newit_counter );
1096 newit_counter++;
1097 } else if( food_contained.is_food() && !food_contained.has_flag( flag_NUTRIENT_OVERRIDE ) ) {
1098 // if a component item has "cooks_like" it will be replaced by that item as a component
1099 for( item &comp : used ) {
1100 // only comestibles have cooks_like. any other type of item will throw an exception, so filter those out
1101 if( comp.is_comestible() && !comp.get_comestible()->cooks_like.is_empty() ) {
1102 comp = item( comp.get_comestible()->cooks_like, comp.birthday(), comp.charges );
1103 }
1104 // If this recipe is cooked or dehydrated, components are no longer raw.
1105 if( should_heat || is_dehydrated ) {
1106 comp.set_flag_recursive( flag_COOKED );
1107 }
1108 }
1109
1110 // use a copy of the used list so that the byproducts don't build up over iterations (#38071)
1111 std::list<item> usedbp = used;
1112 // byproducts get stored as a "component" but with a byproduct flag for consumption purposes
1113 if( making.has_byproducts() ) {
1114 for( item &byproduct : making.create_byproducts( batch_size ) ) {
1115 byproduct.set_flag( flag_BYPRODUCT );
1116 usedbp.push_back( byproduct );
1117 }
1118 }
1119 // store components for food recipes that do not have the override flag
1120 set_components( food_contained.components, usedbp, batch_size, newit_counter );
1121
1122 // store the number of charges the recipe would create with batch size 1.
1123 if( &newit != &food_contained ) { // If a canned/contained item was crafted…
1124 // … the container holds exactly one completion of the recipe, no matter the batch size.
1125 food_contained.recipe_charges = food_contained.charges;
1126 } else { // Otherwise, the item is already stacked so we need to divide by batch size.
1127 newit.recipe_charges = newit.charges / batch_size;
1128 }
1129 newit_counter++;
1130 }
1131
1132 if( food_contained.goes_bad() ) {
1133 food_contained.set_relative_rot( relative_rot );
1134 }
1135
1136 newit.set_owner( p.get_faction()->id );
1137 // If these aren't equal, newit is a container, so finalize its contents too.
1138 if( &newit != &food_contained ) {
1139 food_contained.set_owner( p.get_faction()->id );
1140 }
1141
1142 if( newit.made_of( LIQUID ) ) {
1143 liquid_handler::handle_all_liquid( newit, PICKUP_RANGE );
1144 } else {
1145 set_item_inventory( p, newit );
1146 }
1147 }
1148
1149 if( making.has_byproducts() ) {
1150 std::vector<item> bps = making.create_byproducts( batch_size );
1151 for( auto &bp : bps ) {
1152 if( bp.goes_bad() ) {
1153 bp.set_relative_rot( relative_rot );
1154 }
1155 bp.set_owner( p.get_faction()->id );
1156 bp.inherit_flags( used, making );
1157 if( bp.made_of( LIQUID ) ) {
1158 liquid_handler::handle_all_liquid( bp, PICKUP_RANGE );
1159 } else {
1160 set_item_inventory( p, bp );
1161 }
1162 }
1163 }
1164
1165 p.inv.restack( p );
1166}

1168bool player::can_continue_craft( item &craft )
1169{
1170 if( !craft.is_craft() ) {
1171 debugmsg( "complete_craft() called on non-craft '%s.' Aborting.", craft.tname() );
1172 return false;
1173 }
1174 if( has_trait( trait_DEBUG_HS ) ) {
1175 return true;
1176 }
1177
1178 const recipe &rec = craft.get_making();
1179
1180 const requirement_data continue_reqs = craft.get_continue_reqs();
1181
1182 // Avoid building an inventory from the map if we don't have to, as it is expensive
1183 if( !continue_reqs.is_empty() ) {
1184
1185 std::function<bool( const item & )> filter = rec.get_component_filter();
1186 const std::function<bool( const item & )> no_rotten_filter =
1187 rec.get_component_filter( recipe_filter_flags::no_rotten );
1188 // continue_reqs are for all batches at once
1189 const int batch_size = 1;
1190
1191 if( !continue_reqs.can_make_with_inventory( crafting_inventory(), filter, batch_size ) ) {
1192 std::string buffer = _( "You don't have the required components to continue crafting!" );
1193 buffer += "\n";
1194 buffer += continue_reqs.list_missing();
1195 popup( buffer, PF_NONE );
1196 return false;
1197 }
1198
1199 std::string buffer = _( "Consume the missing components and continue crafting?" );
1200 buffer += "\n";
1201 buffer += continue_reqs.list_all();
1202 if( !query_yn( buffer ) ) {
1203 return false;
1204 }
1205
1206 if( continue_reqs.can_make_with_inventory( crafting_inventory(), no_rotten_filter,
1207 batch_size ) ) {
1208 filter = no_rotten_filter;
1209 } else {
1210 if( !query_yn( _( "Some components required to continue are rotten.\n"
1211 "Continue crafting anyway?" ) ) ) {
1212 return false;
1213 }
1214 }
1215
1216 inventory map_inv;
1217 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1218
1219 std::vector<comp_selection<item_comp>> item_selections;
1220 for( const auto &it : continue_reqs.get_components() ) {
1221 comp_selection<item_comp> is = select_item_component( it, batch_size, map_inv, true, filter );
1222 if( is.use_from == cancel ) {
1223 cancel_activity();
1224 add_msg( _( "You stop crafting." ) );
1225 return false;
1226 }
1227 item_selections.push_back( is );
1228 }
1229 for( const auto &it : item_selections ) {
1230 craft.components.splice( craft.components.end(), consume_items( it, batch_size, filter ) );
1231 }
1232 }
1233
1234 if( !craft.has_tools_to_continue() ) {
1235
1236 const std::vector<std::vector<tool_comp>> &tool_reqs = rec.simple_requirements().get_tools();
1237 const int batch_size = craft.charges;
1238
1239 std::vector<std::vector<tool_comp>> adjusted_tool_reqs;
1240 for( const std::vector<tool_comp> &alternatives : tool_reqs ) {
1241 std::vector<tool_comp> adjusted_alternatives;
1242 for( const tool_comp &alternative : alternatives ) {
1243 tool_comp adjusted_alternative = alternative;
1244 if( adjusted_alternative.count > 0 ) {
1245 adjusted_alternative.count *= batch_size;
1246 // Only for the next 5% progress
1247 adjusted_alternative.count = std::max(
1248 crafting::charges_for_continuing( adjusted_alternative.count ), 1 );
1249 }
1250 adjusted_alternatives.push_back( adjusted_alternative );
1251 }
1252 adjusted_tool_reqs.push_back( adjusted_alternatives );
1253 }
1254
1255 const requirement_data tool_continue_reqs( adjusted_tool_reqs,
1256 std::vector<std::vector<quality_requirement>>(),
1257 std::vector<std::vector<item_comp>>() );
1258
1259 if( !tool_continue_reqs.can_make_with_inventory( crafting_inventory(), return_true<item> ) ) {
1260 std::string buffer = _( "You don't have the necessary tools to continue crafting!" );
1261 buffer += "\n";
1262 buffer += tool_continue_reqs.list_missing();

1264 return false;
1265 }
1266
1267 inventory map_inv;
1268 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1269
1270 std::vector<comp_selection<tool_comp>> new_tool_selections;
1271 for( const std::vector<tool_comp> &alternatives : tool_reqs ) {
1272 comp_selection<tool_comp> selection =
1273 crafting::select_tool_component( alternatives, batch_size, map_inv, this, true, DEFAULT_HOTKEYS,
1274 cost_adjustment::continue_only );
1275 if( selection.use_from == cancel ) {
1276 return false;
1277 }
1278 new_tool_selections.push_back( selection );
1279 }
1280
1281 craft.set_cached_tool_selections( new_tool_selections );
1282 craft.set_tools_to_continue( true );
1283 }

1285 return true;

1319/* selection of component if a recipe requirement has multiple options (e.g. 'duct tap' or 'welder') */
1320comp_selection<item_comp> player::select_item_component( const std::vector<item_comp> &components,
1321 int batch, inventory &map_inv, bool can_cancel,
1322 const std::function<bool( const item & )> &filter, bool player_inv )
1323{
1324 std::vector<item_comp> player_has;
1325 std::vector<item_comp> map_has;
1326 std::vector<item_comp> mixed;
1327
1328 comp_selection<item_comp> selected;
1329
1330 for( const auto &component : components ) {
1331 itype_id type = component.type;
1332 int count = ( component.count > 0 ) ? component.count * batch : std::abs( component.count );
1333
1334 if( item::count_by_charges( type ) && count > 0 ) {
1335 int map_charges = map_inv.charges_of( type, INT_MAX, filter );
1336
1337 // If map has infinite charges, just use them
1338 if( map_charges == item::INFINITE_CHARGES ) {
1339 selected.use_from = use_from_map;
1340 selected.comp = component;
1341 return selected;
1342 }
1343 if( player_inv ) {
1344 int player_charges = charges_of( type, INT_MAX, filter );
1345 bool found = false;
1346 if( player_charges >= count ) {
1347 player_has.push_back( component );
1348 found = true;
1349 }
1350 if( map_charges >= count ) {
1351 map_has.push_back( component );
1352 found = true;
1353 }
1354 if( !found && player_charges + map_charges >= count ) {
1355 mixed.push_back( component );
1356 }
1357 } else {
1358 if( map_charges >= count ) {
1359 map_has.push_back( component );
1360 }
1361 }
1362 } else { // Counting by units, not charges
1363
1364 // Can't use pseudo items as components
1365 if( player_inv ) {
1366 bool found = false;
1367 if( has_amount( type, count, false, filter ) ) {
1368 player_has.push_back( component );
1369 found = true;
1370 }
1371 if( map_inv.has_components( type, count, filter ) ) {
1372 map_has.push_back( component );
1373 found = true;
1374 }
1375 if( !found &&
1376 amount_of( type, false, std::numeric_limits<int>::max(), filter ) +
1377 map_inv.amount_of( type, false, std::numeric_limits<int>::max(), filter ) >= count ) {
1378 mixed.push_back( component );
1379 }
1380 } else {
1381 if( map_inv.has_components( type, count, filter ) ) {
1382 map_has.push_back( component );
1383 }
1384 }
1385 }
1386 }
1387
1388 /* select 1 component to use */
1389 if( player_has.size() + map_has.size() + mixed.size() == 1 ) { // Only 1 choice
1390 if( player_has.size() == 1 ) {
1391 selected.use_from = use_from_player;
1392 selected.comp = player_has[0];
1393 } else if( map_has.size() == 1 ) {
1394 selected.use_from = use_from_map;
1395 selected.comp = map_has[0];
1396 } else {
1397 selected.use_from = use_from_both;
1398 selected.comp = mixed[0];
1399 }
1400 } else if( is_npc() ) {
1401 if( !player_has.empty() ) {
1402 selected.use_from = use_from_player;
1403 selected.comp = player_has[0];
1404 } else if( !map_has.empty() ) {
1405 selected.use_from = use_from_map;
1406 selected.comp = map_has[0];
1407 } else {
1408 debugmsg( "Attempted a recipe with no available components!" );
1409 selected.use_from = cancel;
1410 return selected;
1411 }
1412 } else { // Let the player pick which component they want to use
1413 uilist cmenu;
1414 // Populate options with the names of the items
1415 for( auto &map_ha : map_has ) { // Index 0-(map_has.size()-1)
1416 std::string tmpStr = string_format( _( "%s (%d/%d nearby)" ),
1417 item::nname( map_ha.type ),
1418 ( map_ha.count * batch ),
1419 item::count_by_charges( map_ha.type ) ?
1420 map_inv.charges_of( map_ha.type, INT_MAX, filter ) :
1421 map_inv.amount_of( map_ha.type, false, INT_MAX, filter ) );
1422 cmenu.addentry( tmpStr );
1423 }
1424 for( auto &player_ha : player_has ) { // Index map_has.size()-(map_has.size()+player_has.size()-1)
1425 std::string tmpStr = string_format( _( "%s (%d/%d on person)" ),
1426 item::nname( player_ha.type ),
1427 ( player_ha.count * batch ),
1428 item::count_by_charges( player_ha.type ) ?
1429 charges_of( player_ha.type, INT_MAX, filter ) :
1430 amount_of( player_ha.type, false, INT_MAX, filter ) );
1431 cmenu.addentry( tmpStr );
1432 }
1433 for( auto &component : mixed ) {
1434 // Index player_has.size()-(map_has.size()+player_has.size()+mixed.size()-1)
1435 int available = item::count_by_charges( component.type ) ?
1436 map_inv.charges_of( component.type, INT_MAX, filter ) +
1437 charges_of( component.type, INT_MAX, filter ) :
1438 map_inv.amount_of( component.type, false, INT_MAX, filter ) +
1439 amount_of( component.type, false, INT_MAX, filter );
1440 std::string tmpStr = string_format( _( "%s (%d/%d nearby & on person)" ),
1441 item::nname( component.type ),
1442 component.count * batch,
1443 available );
1444 cmenu.addentry( tmpStr );
1445 }
1446
1447 // Unlike with tools, it's a bad thing if there aren't any components available
1448 if( cmenu.entries.empty() ) {
1449 if( player_inv ) {
1450 if( has_trait( trait_DEBUG_HS ) ) {
1451 selected.use_from = use_from_player;
1452 return selected;
1453 }
1454 }
1455 debugmsg( "Attempted a recipe with no available components!" );
1456 selected.use_from = cancel;
1457 return selected;
1458 }
1459
1460 cmenu.allow_cancel = can_cancel;
1461
1462 // Get the selection via a menu popup
1463 cmenu.title = _( "Use which component?" );
1464 cmenu.query();
1465
1466 if( cmenu.ret < 0 ||
1467 static_cast<size_t>( cmenu.ret ) >= map_has.size() + player_has.size() + mixed.size() ) {
1468 selected.use_from = cancel;
1469 return selected;
1470 }
1471
1472 size_t uselection = static_cast<size_t>( cmenu.ret );
1473 if( uselection < map_has.size() ) {
1474 selected.use_from = usage::use_from_map;
1475 selected.comp = map_has[uselection];
1476 } else if( uselection < map_has.size() + player_has.size() ) {
1477 uselection -= map_has.size();
1478 selected.use_from = usage::use_from_player;
1479 selected.comp = player_has[uselection];
1480 } else {
1481 uselection -= map_has.size() + player_has.size();
1482 selected.use_from = usage::use_from_both;
1483 selected.comp = mixed[uselection];
1484 }
1485 }
1486
1487 return selected;
1488}

1526 const std::function<bool( const item & )> &filter,
1527 const tripoint &origin, int radius )
1528{
1529 std::list<item> ret;
1530
1531 if( has_trait( trait_DEBUG_HS ) ) {
1532 return ret;
1533 }
1534
1535 item_comp selected_comp = is.comp;
1536
1537 const tripoint &loc = origin;
1538 const bool by_charges = item::count_by_charges( selected_comp.type ) && selected_comp.count > 0;
1539 // Count given to use_amount/use_charges, changed by those functions!
1540 int real_count = ( selected_comp.count > 0 ) ? selected_comp.count * batch : std::abs(
1541 selected_comp.count );
1542 // First try to get everything from the map, than (remaining amount) from player
1543 if( is.use_from & use_from_map ) {
1544 if( by_charges ) {
1545 std::list<item> tmp = m.use_charges( loc, radius, selected_comp.type, real_count, filter );
1546 ret.splice( ret.end(), tmp );
1547 } else {
1548 std::list<item> tmp = g->m.use_amount( loc, radius, selected_comp.type, real_count, filter );
1549 remove_ammo( tmp, *this );
1550 ret.splice( ret.end(), tmp );
1551 }
1552 }
1553 if( is.use_from & use_from_player ) {
1554 if( by_charges ) {
1555 std::list<item> tmp = use_charges( selected_comp.type, real_count, filter );
1556 ret.splice( ret.end(), tmp );
1557 } else {
1558 std::list<item> tmp = use_amount( selected_comp.type, real_count, filter );
1559 remove_ammo( tmp, *this );
1560 ret.splice( ret.end(), tmp );
1561 }
1562 }

1564 if( by_charges && ret.size() > 1 ) {
1565 std::list<item>::iterator b = ret.begin();
1566 b++;

1568 ret.front().charges += b->charges;
1569 b = ret.erase( b );
1570 }
1571 }
1572 lastconsumed = selected_comp.type;
1573 empty_buckets( *this );
1574 return ret;
1575}

1604static std::vector<avail_tool_comp>
1605find_tool_component( const Character *player_with_inv, const std::vector<tool_comp> &tools,
1606 int batch,
1607 const inventory &map_inv, cost_adjustment charge_mod )
1608{
1609 std::vector<avail_tool_comp> available_tools;
1610
1611 auto calc_charges = [&]( const tool_comp & t ) {
1612 const int full_craft_charges = std::max( 1, t.count * batch );
1613 switch( charge_mod ) {
1614 case cost_adjustment::none:
1615 return std::make_pair( charges_for_complete( full_craft_charges ),
1616 charges_for_complete( full_craft_charges ) );
1617 case cost_adjustment::start_only:
1618 return std::make_pair( charges_for_starting( full_craft_charges ),
1619 charges_for_complete( full_craft_charges ) );
1620 case cost_adjustment::continue_only:
1621 return std::make_pair( charges_for_continuing( full_craft_charges ),
1622 charges_for_complete( full_craft_charges ) );
1623 }
1624
1625 debugmsg( "Invalid tool_charge_mod" );
1626 return std::make_pair( INT_MAX, INT_MAX );
1627 };
1628
1629 bool found_nocharge = false;
1630 // Use charges of any tools that require charges used
1631 for( auto it = tools.begin(); it != tools.end() && !found_nocharge; ++it ) {
1632 itype_id type = it->type;
1633 if( it->count > 0 ) {
1634 const std::pair<int, int> &expected_count = calc_charges( *it );
1635 const int count = expected_count.first;
1636 const int ideal = expected_count.second;
1637 if( player_with_inv ) {
1638 if( player_with_inv->has_charges( type, count ) ) {
1639 int total_charges = player_with_inv->charges_of( type );
1640 comp_selection<tool_comp> sel( use_from_player, *it );
1641 available_tools.emplace_back( sel, total_charges, ideal );
1642 }
1643 }
1644 if( map_inv.has_charges( type, count ) ) {
1645 int total_charges = map_inv.charges_of( type );
1646 comp_selection<tool_comp> sel( use_from_map, *it );
1647 available_tools.emplace_back( sel, total_charges, ideal );
1648 }
1649 } else if( ( player_with_inv && player_with_inv->has_amount( type, 1 ) )
1650 || map_inv.has_tools( type, 1 ) ) {
1651 comp_selection<tool_comp> sel( use_from_none, *it );
1652 available_tools.emplace_back( sel, 0, 0 );
1653 }
1654 }
1655
1656 std::sort( available_tools.begin(), available_tools.end(),
1657 []( const avail_tool_comp & lhs, const avail_tool_comp & rhs ) {
1658 if( lhs.comp.use_from == use_from_none && rhs.comp.use_from != use_from_none ) {
1659 return true;
1660 }
1661 if( rhs.comp.use_from == use_from_none ) {
1662 return false;
1663 }
1664 if( lhs.charges >= lhs.ideal && rhs.charges < rhs.ideal ) {
1665 return true;
1666 }
1667 if( lhs.charges < lhs.ideal && rhs.charges >= rhs.ideal ) {
1668 return false;
1669 }
1670 // We want "bigger" tools first because those are likely to be vehicle mounted
1671 return static_cast<float>( lhs.ideal ) / lhs.charges > static_cast<float>
1672 ( rhs.ideal ) / rhs.charges;
1673 } );
1674 return available_tools;
1675}

1762bool player::craft_consume_tools( item &craft, int mulitplier, bool start_craft )
1763{
1764 if( !craft.is_craft() ) {
1765 debugmsg( "craft_consume_tools() called on non-craft '%s.' Aborting.", craft.tname() );
1766 return false;
1767 }
1768 if( has_trait( trait_DEBUG_HS ) ) {
1769 return true;
1770 }
1771 if( start_craft && mulitplier > 1 ) {
1772 debugmsg( "start_craft is true, but multiplier is %d > 1", mulitplier );
1773 return false;
1774 }
1775
1776 const auto calc_charges = [&craft, &start_craft, &mulitplier]( int charges ) {
1777 int ret = charges;
1778
1779 if( charges <= 0 ) {
1780 return ret;
1781 }
1782
1783 // Account for batch size
1784 int full_cost = charges * craft.charges;
1785
1786 if( start_craft ) {
1787 return crafting::charges_for_starting( full_cost );
1788 } else {
1789 // In case more than 5% progress was accomplished in one turn
1790 return crafting::charges_for_continuing( full_cost ) * mulitplier;
1791 }
1792 };
1793
1794 // First check if we still have our cached selections
1795 const std::vector<comp_selection<tool_comp>> &cached_tool_selections =
1796 craft.get_cached_tool_selections();
1797
1798 inventory map_inv;
1799 map_inv.form_from_map( pos(), PICKUP_RANGE, this );
1800
1801 for( const comp_selection<tool_comp> &tool_sel : cached_tool_selections ) {
1802 itype_id type = tool_sel.comp.type;
1803 if( tool_sel.comp.count > 0 ) {
1804 const int count = calc_charges( tool_sel.comp.count );
1805 switch( tool_sel.use_from ) {
1806 case use_from_player:
1807 if( !has_charges( type, count ) ) {
1808 add_msg_player_or_npc(
1809 _( "You have insufficient %s charges and can't continue crafting" ),
1810 _( "<npcname> has insufficient %s charges and can't continue crafting" ),
1811 item::nname( type ) );
1812 craft.set_tools_to_continue( false );
1813 return false;
1814 }
1815 break;
1816 case use_from_map:
1817 if( !map_inv.has_charges( type, count ) ) {
1818 add_msg_player_or_npc(
1819 _( "You have insufficient %s charges and can't continue crafting" ),
1820 _( "<npcname> has insufficient %s charges and can't continue crafting" ),
1821 item::nname( type ) );
1822 craft.set_tools_to_continue( false );
1823 return false;
1824 }
1825 break;
1826 case use_from_both:
1827 case use_from_none:
1828 case cancel:
1829 case num_usages:
1830 break;
1831 }

1887ret_val<bool> crafting::can_disassemble( const Character &who, const item &obj,
1888 const inventory &inv )
1889{
1890 const auto &r = recipe_dictionary::get_uncraft( obj.typeId() );
1891
1892 if( !r || obj.has_flag( flag_ETHEREAL_ITEM ) ) {
1893 return ret_val<bool>::make_failure( _( "You cannot disassemble this." ) );
1894 }
1895
1896 // check sufficient light
1897 if( lighting_crafting_speed_multiplier( who, r ) == 0.0f ) {
1898 return ret_val<bool>::make_failure( _( "You can't see to craft!" ) );
1899 }
1900
1901 // refuse to disassemble rotten items
1902 const item *food = obj.get_food();
1903 if( ( obj.goes_bad() && obj.rotten() ) || ( food && food->goes_bad() && food->rotten() ) ) {
1904 return ret_val<bool>::make_failure( _( "It's rotten, I'm not taking that apart." ) );
1905 }
1906
1907 // refuse to disassemble items containing monsters/pets
1908 std::string monster = obj.get_var( "contained_name" );
1909 if( !monster.empty() ) {
1910 return ret_val<bool>::make_failure( _( "You must remove the %s before you can disassemble this." ),
1911 monster );
1912 }
1913
1914 if( obj.count_by_charges() ) {
1915 int batch_size = r.disassembly_batch_size();
1916 if( obj.charges < batch_size ) {
1917 auto msg = vgettext( "You need at least %d charge of %s.",
1918 "You need at least %d charges of %s.", batch_size );
1919 return ret_val<bool>::make_failure( msg, batch_size, obj.tname() );
1920 }
1921 }
1922 const auto &dis = r.disassembly_requirements();
1923
1924 for( const auto &opts : dis.get_qualities() ) {
1925 for( const auto &qual : opts ) {
1926 if( !qual.has( inv, return_true<item> ) ) {
1927 // Here should be no dot at the end of the string as 'to_string()' provides it.
1928 return ret_val<bool>::make_failure( _( "You need %s" ), qual.to_string() );
1929 }
1930 }
1931 }
1932
1933 for( const auto &opts : dis.get_tools() ) {
1934 const bool found = std::any_of( opts.begin(), opts.end(),
1935 [&]( const tool_comp & tool ) {
1936 return ( tool.count <= 0 && inv.has_tools( tool.type, 1 ) ) ||
1937 ( tool.count > 0 && inv.has_charges( tool.type, tool.count ) );
1938 } );
1939
1940 if( !found ) {
1941 const tool_comp &tool_required = opts.front();
1942 if( tool_required.count <= 0 ) {

1944 item::nname( tool_required.type ) );
1945 } else {
1946 //~ %1$s: tool name, %2$d: needed charges
1947 return ret_val<bool>::make_failure( vgettext( "You need a %1$s with %2$d charge.",
1948 "You need a %1$s with %2$d charges.", tool_required.count ),
1949 item::nname( tool_required.type ),
1950 tool_required.count );
1951 }
1952 }
1953 }
1954
1955 return ret_val<bool>::make_success();
1956}
