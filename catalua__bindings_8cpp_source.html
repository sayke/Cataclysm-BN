34std::string cata::detail::fmt_lua_va( sol::variadic_args va )
35{
36 lua_State *L = va.lua_state();
37 sol::state_view lua( L );
38
39 std::string msg;
40 for( auto it : va ) {
41 msg += lua["tostring"]( it );
42 }
43
44 return msg;
45}

55struct item_stack_lua_it_state {
56 using it_t = item_stack::iterator;
57 it_t it;
58 it_t last;
59
60 item_stack_lua_it_state( item_stack &stk )
61 : it( stk.begin() ), last( stk.end() ) {
62 }
63};
64
65static std::tuple<sol::object, sol::object>
66item_stack_lua_next(
67 sol::user<item_stack_lua_it_state &> user_it_state,
68 sol::this_state l )
69{
70 // this gets called
71 // to start the first iteration, and every
72 // iteration there after
73
74 // the state you passed in item_stack_lua_pairs is argument 1
75 // the key value is argument 2, but we do not
76 // care about the key value here
77 item_stack_lua_it_state &it_state = user_it_state;
78 auto &it = it_state.it;
79 if( it == it_state.last ) {
80 // return nil to signify that
81 // there's nothing more to work with.
82 return std::make_tuple( sol::object( sol::lua_nil ),
83 sol::object( sol::lua_nil ) );
84 }
85 item *elem = &*it;
86 // 2 values are returned (pushed onto the stack):
87 // the key and the value
88 // the state is left alone
89 auto r = std::make_tuple(
90 sol::object( l, sol::in_place, it ),
91 sol::object( l, sol::in_place, elem ) );
92 // the iterator must be moved forward one before we return
93 std::advance( it, 1 );
94 return r;
95}
96
97static auto item_stack_lua_pairs( item_stack &stk )
98{
99 // pairs expects 3 returns:
100 // the "next" function on how to advance,
101 // the "table" itself or some state,
102 // and an initial key value (can be nil)
103
104 // prepare our state
105 item_stack_lua_it_state it_state( stk );
106 // sol::user is a space/time optimization over regular
107 // usertypes, it's incompatible with regular usertypes and
108 // stores the type T directly in lua without any pretty
109 // setup saves space allocation and a single dereference
110 return std::make_tuple( &item_stack_lua_next,
111 sol::user<item_stack_lua_it_state>( std::move( it_state ) ),
112 sol::lua_nil );
113}

558template<typename E>
559void reg_enum( sol::state &lua )
560{
561 // Sol2 has new_enum<E>(...) function, but it needs to know all value-string
562 // pairs at compile time, so we can't use it with io::enum_to_string.
563 //
564 // As such, hack it by creating read-only table.
565
566 luna::userenum<E> et = luna::begin_enum<E>( lua );
567
568 using Int = std::underlying_type_t<E>;
569 constexpr Int max = static_cast<Int>( enum_traits<E>::last );
570
571 for( Int i = 0; i < max; ++i ) {
572 E e = static_cast<E>( i );
573 std::string key = io::enum_to_string<E>( e );
574 luna::add_val( et, key, e );
575 }
576
577 luna::finalize_enum( et );
578}
