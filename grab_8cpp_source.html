34// vehicle movement: strength check. very strong humans can move about 2,000 kg in a wheelbarrow.

46// determine movecost for terrain touching wheels
47auto get_grabbed_vehicle_movecost( vehicle *veh ) -> int
48{
49 const int str_req = base_str_req( veh );
50 const auto &map = get_map();
51 const tripoint &vehpos = veh->global_pos3();
52
53 static const auto get_wheel_pos = [&]( const int p ) {
54 return vehpos + veh->part( p ).precalc[0];
55 };
56
57 const auto &wheel_indices = veh->wheelcache;
58 return std::accumulate( wheel_indices.begin(), wheel_indices.end(), 0,
59 [&]( const int sum, const int p ) {
60 const tripoint wheel_pos = get_wheel_pos( p );
61 const int mapcost = map.move_cost( wheel_pos, veh );
62 const int movecost = str_req / static_cast<int>( wheel_indices.size() ) * mapcost;
63
64 return sum + movecost;
65 } );
66}

69auto get_effective_wheels( vehicle *veh ) -> int

76// very strong humans can move about 2,000 kg in a wheelbarrow.
77auto get_vehicle_str_requirement( vehicle *veh ) -> int
78{
79 if( !veh->valid_wheel_config() ) {
80 return base_str_req( veh ) * 10;
81 }
82
83 //if vehicle is rollable we modify str_req based on a function of movecost per wheel.
84 const int all_movecost = get_grabbed_vehicle_movecost( veh );
85 // off-road coefficient (always 1.0 on a road, as low as 0.1 off road.)
86 const float traction = veh->k_traction(
87 get_map().vehicle_wheel_traction( *veh ) );
88 return int { 1 + all_movecost / get_effective_wheels( veh ) } / traction;
89}
