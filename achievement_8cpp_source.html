24// Some details about how achievements work
25// ========================================
26//
27// Achievements are built on the stats_tracker, which is in turn built on the
28// event_bus. Each of these layers involves subscription / callback style
29// interfaces, so the code flow may now be obvious. Here's a quick outline of
30// the execution flow to help clarify how it all fits together.
31//
32// * Various core game code paths generate events via the event bus.
33// * The stats_tracker subscribes to the event bus, and receives these events.
34// * Events contribute to event_multisets managed by the stats_tracker.
35// * (In the docs, these event_multisets are described as "event streams").
36// * (Optionally) event_transformations transform these event_multisets into
37// other event_multisets based on json-defined transformation rules. These
38// are also managed by stats_tracker.
39// * event_statistics monitor these event_multisets and summarize them into
40// single values. These are also managed by stats_tracker.
41// * Each achievement requirement has a corresponding requirement_watcher which
42// is alerted to statistic changes via the stats_tracker's watch interface.
43// * Each requirement_watcher notifies its achievement_tracker (of which there
44// is one per achievement) of the requirement's status on each change to a
45// statistic.
46// * The achievement_tracker keeps track of which requirements are currently
47// satisfied and which are not.
48// * When all the requirements are satisfied, the achievement_tracker tells the
49// achievement_tracker (only one of these exists per game).
50// * The achievement_tracker calls the achievement_attained_callback it was
51// given at construction time. This hooks into the actual game logic (e.g.
52// telling the player they just got an achievement).

280achievement_completion time_req_completed( const achievement &ach )
281{
282 if( !ach.time_constraint() ) {
283 return achievement_completion::completed;
284 }
285
286 const std::optional<achievement::time_bound> &time_req = ach.time_constraint();
287
288 time_point now = calendar::turn;
289 switch( time_req->comparison() ) {
290 case achievement_comparison::less_equal:
291 if( now <= time_req->target() ) {
292 return achievement_completion::completed;
293 } else {
294 return achievement_completion::failed;
295 }
296 case achievement_comparison::greater_equal:
297 if( now >= time_req->target() ) {
298 return achievement_completion::completed;
299 } else {
300 return achievement_completion::pending;
301 }
302 case achievement_comparison::anything:
303 return achievement_completion::completed;
304 case achievement_comparison::last:
305 break;
306 }
307 debugmsg( "Invalid achievement_comparison" );
308 abort();

312{

314 ach.kill_requirements();

316 ach.species_kill_requirements();
317 if( kill_reqs.empty() && species_kill_reqs.empty() ) {
318 return achievement_completion::completed;
319 }
320
321 achievement_completion satisfied = achievement_completion::completed;
322 for( const auto& [m_id, pair] : kill_reqs ) {
323 if( satisfied == achievement_completion::failed ) {
324 break;
325 }
326 auto& [comp, count] = pair;
327 int kill_count = m_id == mtype_id::NULL_ID() ? kt.monster_kill_count() :
328 kt.kill_count( m_id );
329 if( !ach_compare( comp, count, kill_count ) ) {
330 switch( comp ) {
331 case achievement_comparison::greater_equal:
332 // Still might reach target, check for other fail conditions.
333 satisfied = achievement_completion::pending;
334 break;
335 case achievement_comparison::less_equal:
336 // Can't reduce kill count.
337 satisfied = achievement_completion::failed;
338 break;
339 default:
340 break;
341 }
342 }
343 }

345 if( satisfied == achievement_completion::failed ) {
346 break;
347 }
348 auto& [comp, count] = pair;
349 int kill_count = kt.kill_count( fac_id );
350 if( !ach_compare( comp, count, kill_count ) ) {
351 switch( comp ) {
352 case achievement_comparison::greater_equal:
353 // Still might reach target, check for other fail conditions.
354 satisfied = achievement_completion::pending;
355 break;
356 case achievement_comparison::less_equal:
357 // Can't reduce kill count.
358 satisfied = achievement_completion::failed;
359 break;
360 default:
361 break;
362 }

583std::string achievement::time_bound::time_ui_text( const achievement_completion comp ) const
584{
585
586 time_point now = calendar::turn;
587

589

591 switch( e ) {
592 case epoch::cataclysm:
593 return _( "time of cataclysm" );
594 case epoch::game_start:
595 return _( "start of game" );
596 case epoch::last:

609 case achievement_completion::completed:

612 return string_format( _( "Within %s of %s (%s remaining)" ),

614 to_string( target() - now ) );
615 case achievement_comparison::greater_equal:
616 return string_format( _( "At least %s from %s (passed)" ),
617 to_string( period_ ), translate_epoch( epoch_ ) );
618 case achievement_comparison::anything:
619 return std::string();
620 case achievement_comparison::last:
621 break;
622 }
623 debugmsg( "Invalid achievement_comparison" );
624 abort();
625 case achievement_completion::failed:
626 return string_format( _( "Within %s of %s (passed)" ),
627 to_string( period_ ), translate_epoch( epoch_ ) );
628 case achievement_completion::last:
629 break;
630 }
631 debugmsg( "Invalid achievement_completion" );
632 abort();
633 }
634 ();
635
636 return colorize( message, c );

685{

687 if( !a.is_valid() ) {
688 debugmsg( "Achievement %s specifies hidden_by achievement %s, but the latter does not "
689 "exist.", id.str(), a.str() );
690 }
691 }
692 if( time_constraint_ ) {
693 time_constraint_->check( id );
694 }
695 if( !skill_requirements_.empty() ) {
696 for( const auto& [sk_id, pair] : skill_requirements_ ) {
697 auto &&[comp, level] = pair;

700 } else if( comp == achievement_comparison::anything ) {

705 }

713 } else if( comp == achievement_comparison::anything ) {
714 debugmsg( "Achievement %s specifies invalid comparator for kill requirement %s.", id.str(),
715 m_id.c_str() );

717 debugmsg( "Achievement %s has negative count for kill requirement %s.", id.str(), m_id.c_str() );
718 }
719 }
720 }
721 if( !species_kill_requirements_.empty() ) {
722 for( const auto& [fac_id, pair] : species_kill_requirements_ ) {
723 auto &&[comp, count] = pair;
724 if( !fac_id.is_valid() || fac_id.is_null() ) {
725 debugmsg( "Achievement %s specifies invalid kill requirement %s.", id.str(), fac_id.c_str() );
726 } else if( comp == achievement_comparison::anything ) {
727 debugmsg( "Achievement %s specifies invalid comparator for kill requirement %s.", id.str(),
728 fac_id.c_str() );
729 } else if( count < 0 ) {
730 debugmsg( "Achievement %s has negative count for kill requirement %s.", id.str(), fac_id.c_str() );
731 }
732 }
733 }
734 for( const achievement_requirement &req : requirements_ ) {
735 req.check( id );
736 }

740 const cata_variant &current_value )
741{
742 bool is_satisfied = req.satisifed_by( current_value );
743 nc_color c = is_satisfied ? c_green : c_yellow;
744 int current = current_value.get<int>();
745 int target;
746 std::string result;
747 if( req.comparison == achievement_comparison::anything ) {
748 target = 1;
749 result = string_format( _( "Triggered by " ) );

751 target = req.target;
752 result = string_format( _( "%s/%s " ), current, target );
753 }
754 result += req.statistic->description().translated( target );
755 return colorize( result, c );

869achievement_tracker::achievement_tracker( const achievement &a, achievements_tracker &tracker,
870 stats_tracker &stats ) :
871 achievement_( &a ),
872 tracker_( &tracker )
873{
874 for( const achievement_requirement &req : a.requirements() ) {
875 watchers_.push_back( std::make_unique<requirement_watcher>( *this, req, stats ) );
876 }
877
878 for( const std::unique_ptr<requirement_watcher> &watcher : watchers_ ) {
879 bool is_satisfied = watcher->is_satisfied( stats );
880 sorted_watchers_[is_satisfied].insert( watcher.get() );
881 }
882}
883
884void achievement_tracker::set_requirement( requirement_watcher *watcher, bool is_satisfied )
885{
886 if( sorted_watchers_[is_satisfied].insert( watcher ).second ) {
887 // Remove from other; check for completion.
888 sorted_watchers_[!is_satisfied].erase( watcher );
889 assert( sorted_watchers_[0].size() + sorted_watchers_[1].size() == watchers_.size() );
890 }
891
892 achievement_completion time_comp = time_req_completed( *achievement_ );
893 achievement_completion skill_comp = skill_req_completed( *achievement_ );
894 achievement_completion kill_comp = kill_req_completed( *achievement_, *tracker_->kills() );
895 bool all_clear = time_comp == achievement_completion::completed &&
896 skill_comp == achievement_completion::completed && kill_comp == achievement_completion::completed;
897
898 if( sorted_watchers_[false].empty() && all_clear ) {
899 // report_achievement can result in this being deleted, so it must be
900 // the last thing in the function
901 tracker_->report_achievement( achievement_, achievement_completion::completed );
902 return;
903 }
904
905 if( time_comp == achievement_completion::failed || kill_comp == achievement_completion::failed ||
906 ( !is_satisfied && watcher->requirement().becomes_false ) ) {
907 // report_achievement can result in this being deleted, so it must be
908 // the last thing in the function
909 tracker_->report_achievement( achievement_, achievement_completion::failed );
910 }
911}

916 skill_req_completed( *achievement_ ) == achievement_completion::failed ||
917 kill_req_completed( *achievement_, *tracker_->kills() ) == achievement_completion::failed;
918 return failed;
919}
