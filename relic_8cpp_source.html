276bool check_recharge_reqs( const item &itm, const relic_recharge &rech, const Character &carrier )
277{
278 switch( rech.req ) {
279 case relic_recharge_req::none: {
280 return true;
281 }
282 case relic_recharge_req::equipped: {
283 if( itm.is_armor() ) {
284 return carrier.is_wearing( itm );
285 } else {
286 return carrier.is_wielding( itm );
287 }
288 }
289 case relic_recharge_req::close_to_skin: {
290 if( itm.is_armor() ) {
291 if( !carrier.is_wearing( itm ) ) {
292 return false;
293 }
294 for( const bodypart_id &bp : carrier.get_all_body_parts() ) {
295 if( !itm.covers( bp ) ) {
296 continue;
297 }
298 bool this_bp_good = true;
299 for( const item &wi : carrier.worn ) {
300 if( wi.get_coverage() == 0 ) {
301 continue;
302 }
303 if( &wi == &itm ) {
304 break;
305 } else if( wi.covers( bp ) ) {
306 this_bp_good = false;
307 break;
308 }
309 }
310 if( this_bp_good ) {
311 return true;
312 }
313 }
314 return false;
315 } else {
316 if( !carrier.is_wielding( itm ) ) {
317 return false;
318 }
319 bool hand_l_ok = true;
320 bool hand_r_ok = true;
321 for( const item &wi : carrier.worn ) {
322 if( wi.get_coverage() == 0 ) {
323 continue;
324 }
325 if( wi.covers( body_part_hand_l ) ) {
326 hand_l_ok = false;
327 }
328 if( wi.covers( body_part_hand_r ) ) {
329 hand_r_ok = false;
330 }
331 }
332 return hand_l_ok || hand_r_ok;
333 }
334 }
335 case relic_recharge_req::sleep: {
336 return carrier.has_effect( effect_sleep );
337 }
338 case relic_recharge_req::rad: {
339 return get_map().get_radiation( carrier.pos() ) > 0 || carrier.get_rad() > 0;
340 }
341 case relic_recharge_req::wet: {
342 bool has_wet = std::any_of( carrier.body_wetness.begin(), carrier.body_wetness.end(),
343 []( const int w ) {
344 return w != 0;
345 } );
346 if( has_wet ) {
347 return true;
348 } else {
349 const weather_type &wt = get_weather().weather_id.obj();
350 if( !wt.rains || wt.acidic || wt.precip == precip_class::none ) {
351 return false;
352 }
353 return get_map().is_outside( carrier.pos() );
354 }
355 }
356 case relic_recharge_req::sky: {
357 return carrier.posz() > 0;
358 }
359 default: {
360 std::abort();
361 }
362 }
363 cata::unreachable();
364}

366bool process_recharge_entry( item &itm, const relic_recharge &rech, Character &carrier )
367{
368 if( !calendar::once_every( rech.interval ) ) {
369 return false;
370 }
371 if( !check_recharge_reqs( itm, rech, carrier ) ) {
372 return false;
373 }
374 map &here = get_map();
375 switch( rech.type ) {
376 case relic_recharge_type::time: {
377 break;
378 }
379 case relic_recharge_type::solar: {
380 if( !g->is_in_sunlight( carrier.pos() ) ) {
381 return false;
382 }
383 break;
384 }
385 case relic_recharge_type::pain: {
386 carrier.add_msg_if_player( m_bad, _( "You suddenly feel sharp pain for no reason." ) );
387 carrier.mod_pain_noresist( rng( rech.intensity_min, rech.intensity_max ) );
388 break;
389 }
390 case relic_recharge_type::hp: {
391 carrier.add_msg_if_player( m_bad, _( "You feel your body decaying." ) );
392 carrier.hurtall( rng( rech.intensity_min, rech.intensity_max ), nullptr );
393 break;
394 }
395 case relic_recharge_type::fatigue: {
396 carrier.add_msg_if_player( m_bad, _( "You feel fatigue seeping into your body." ) );
397 carrier.mod_fatigue( rng( rech.intensity_min, rech.intensity_max ) );
398 carrier.mod_stamina( -100 * rng( rech.intensity_min, rech.intensity_max ) );
399 break;
400 }
401 case relic_recharge_type::field: {
402 bool consumed = false;
403 for( const tripoint &dest : here.points_in_radius( carrier.pos(), 1 ) ) {
404 field_entry *field_at = here.field_at( dest ).find_field( rech.field_type );
405 if( !field_at ) {
406 continue;
407 }
408 int int_here = field_at->get_field_intensity();
409 if( int_here >= rech.intensity_min && int_here <= rech.intensity_max ) {
410 here.remove_field( dest, rech.field_type );
411 consumed = true;
412 break;
413 }
414 }
415 if( !consumed ) {
416 return false;
417 }
418 break;
419 }
420 case relic_recharge_type::trap: {
421 bool consumed = false;
422 for( const tripoint &dest : here.points_in_radius( carrier.pos(), 1 ) ) {
423 if( here.tr_at( dest ).id == rech.trap_type ) {
424 here.remove_trap( dest );
425 consumed = true;
426 }
427 }
428 if( !consumed ) {
429 return false;
430 }
431 break;
432 }
433 default: {
434 std::abort();
435 }
436 }
437 // If relic has a valid ammo type, make sure the first charge loaded isn't a "none"
438 bool was_zero = itm.charges == 0;
439 itm.charges = clamp( itm.charges + rech.rate, 0, itm.ammo_capacity() );
440 if( was_zero && !itm.ammo_types().empty() ) {
441 itm.ammo_set( itm.ammo_default(), itm.charges );
442 }
443 if( rech.message ) {
444 carrier.add_msg_if_player( _( *rech.message ) );
445 }
446 return true;
447}
