126SDL_Color color_pixel_sepia( const SDL_Color &color )
127{
128 if( is_black( color ) ) {
129 return color;
130 }
131
132 /*
133 * Objective is to provide a gradient between two color points
134 * (sepia_dark and sepia_light) based on the grayscale value.
135 * This presents an effect intended to mimic a faded sepia photograph.
136 */
137
138 const SDL_Color sepia_dark = { 39, 23, 19, color.a};
139 const SDL_Color sepia_light = { 241, 220, 163, color.a};
140
141 const Uint8 av = average_pixel_color( color );
142 const float gammav = 1.6;
143 const float pv = av / 255.0;
144 const Uint8 finalv =
145 std::min( static_cast<int>( std::round( std::pow( pv, gammav ) * 150 ) ), 100 );
146
147 return mix_colors( sepia_dark, sepia_light, finalv );
148}

172std::vector<SDL_Color> color_linear_interpolate( const SDL_Color &start_color,
173 const SDL_Color &end_color, const unsigned additional_steps )
174{
175 const unsigned steps = additional_steps + 2;
176 std::vector<SDL_Color> colors;
177 colors.reserve( steps );
178 colors.push_back( start_color );
179
180 const float step = 1.0f / ( steps - 1 );
181 for( unsigned i = 0; i < steps - 2; i++ ) {
182 const float percent = ( i + 1 ) * step;
183 const Uint8 r = static_cast<Uint8>( start_color.r + static_cast<Uint8>( percent *
184 ( end_color.r - start_color.r ) ) );
185 const Uint8 g = static_cast<Uint8>( start_color.g + static_cast<Uint8>( percent *
186 ( end_color.g - start_color.g ) ) );
187 const Uint8 b = static_cast<Uint8>( start_color.b + static_cast<Uint8>( percent *
188 ( end_color.b - start_color.b ) ) );
189 colors.push_back( { r, g, b, 255 } );
190 }
191 colors.push_back( end_color );
192 return colors;
193}

204SDL_Rect fit_rect_inside( const SDL_Rect &inner, const SDL_Rect &outer )
205{
206 const float inner_ratio = static_cast<float>( inner.w ) / inner.h;
207 const float outer_ratio = static_cast<float>( outer.w ) / outer.h;
208 const float factor = inner_ratio > outer_ratio
209 ? static_cast<float>( outer.w ) / inner.w
210 : static_cast<float>( outer.h ) / inner.h;
211
212 const int w = factor * inner.w;
213 const int h = factor * inner.h;
214 const point p( outer.x + ( outer.w - w ) / 2, outer.y + ( outer.h - h ) / 2 );
215
216 return SDL_Rect{ p.x, p.y, w, h };
217}
