366template<typename Class, typename ConstructorScheme, typename Bases>
367sol::usertype<Class> new_usertype(
368 sol::state_view &lua,
369 Bases &&bases,
370 ConstructorScheme &&constructor
371)
372{
373 static_assert( detail::luna_traits<Class>::impl, "Type must implement luna_traits<T>" );
374
375 // Ensure global doctable exists
376 sol::table global_dt = detail::get_global_doctable( lua );
377
378 constexpr std::string_view name = detail::luna_traits<Class>::name;
379
380 // Register Sol usertype
381 sol::usertype<Class> ut;
382 using BasesBare = detail::remove_cv_ref_t<Bases>;
383 if constexpr( std::is_same_v<BasesBare, no_bases_t> ) {
384 ut = lua.new_usertype<Class>( name, constructor );
385 } else {
386 ut = lua.new_usertype<Class>( name, constructor, sol::base_classes, bases );
387 }
388
389 // Create doctable for this type
390 sol::table type_dt = lua.create_table();
391 global_dt[detail::KEY_TYPES][name] = type_dt;
392
393 // Link relevant Sol usertype in docs
394 type_dt[detail::KEY_TYPE_IMPL] = ut;
395
396 // Doc comment
397 detail::add_comment( type_dt, detail::KEY_USERTYPE_COMMENT );
398
399 // Init members table
400 type_dt[detail::KEY_MEMBER] = std::vector<sol::object>();
401
402 // Document constructors (or lack thereof)
403 detail::doc_constructors( type_dt, constructor );
404
405 // Document bases (or lack thereof)
406 detail::doc_bases( type_dt, bases );
407
408 // Add helper method to get name under which usertype is bound in Lua
409 ut[detail::KEY_GET_TYPE] = []() -> std::string_view {
410 return detail::luna_traits<Class>::name;
411 };
412
413 return ut;
414}
