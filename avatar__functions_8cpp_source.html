514std::pair<int, int> gunmod_installation_odds( const avatar &you, const item &gun,
515 const item &mod )
516{
517 // Mods with INSTALL_DIFFICULT have a chance to fail, potentially damaging the gun
518 if( !mod.has_flag( "INSTALL_DIFFICULT" ) || you.has_trait( trait_DEBUG_HS ) ) {
519 return std::make_pair( 100, 0 );
520 }
521
522 int roll = 100; // chance of success (%)
523 int risk = 0; // chance of failure (%)
524 int chances = 1; // start with 1 in 6 (~17% chance)
525
526 for( const auto &e : mod.type->min_skills ) {
527 // gain an additional chance for every level above the minimum requirement
528 skill_id sk = e.first == skill_weapon ? gun.gun_skill() : e.first;
529 chances += std::max( you.get_skill_level( sk ) - e.second, 0 );
530 }
531 // cap success from skill alone to 1 in 5 (~83% chance)
532 roll = std::min( static_cast<double>( chances ), 5.0 ) / 6.0 * 100;
533 // focus is either a penalty or bonus of at most +/-10%
534 roll += ( std::max( std::min( you.focus_pool, 140 ), 60 ) - 100 ) / 4;
535 // dexterity and intelligence give +/-2% for each point above or below 12
536 roll += ( you.get_dex() - 12 ) * 2;
537 roll += ( you.get_int() - 12 ) * 2;
538 // each level of damage to the base gun reduces success by 10%
539 roll -= std::max( gun.damage_level( 4 ), 0 ) * 10;
540 roll = std::min( std::max( roll, 0 ), 100 );
541
542 // risk of causing damage on failure increases with less durable guns
543 risk = ( 100 - roll ) * ( ( 10.0 - std::min( gun.type->gun->durability, 9 ) ) / 10.0 );
544
545 return std::make_pair( roll, risk );
546}

649bool unload_item( avatar &you, item_location loc )
650{
651 item &it = *loc.get_item();
652 //Give the player the same options as when attempting to eat food that doesn't belong to them, bomb out if they say no.
653 if( !query_consume_ownership( it, you ) ) {
654 return false;
655 }
656 // Unload a container consuming moves per item successfully removed
657 if( it.is_container() || it.is_bandolier() || it.type->can_use( "holster" ) ) {
658 if( it.contents.empty() ) {
659 add_msg( m_info, _( "The %s is already empty!" ), it.tname() );
660 return false;
661 }
662 if( !it.can_unload_liquid() ) {
663 add_msg( m_info, _( "The liquid can't be unloaded in its current state!" ) );
664 return false;
665 }
666
667 bool changed = false;
668 for( item *contained : it.contents.all_items_top() ) {
669 int old_charges = contained->charges;
670 const bool consumed = add_or_drop_with_msg( you, *contained, true );
671 changed = changed || consumed || contained->charges != old_charges;
672 if( consumed ) {
673 you.mod_moves( -you.item_handling_cost( *contained ) );
674 it.remove_item( *contained );
675 }
676 }
677
678 if( changed ) {
679 it.on_contents_changed();
680 }
681 return true;
682 }
683
684 // If item can be unloaded more than once (currently only guns) prompt user to choose
685 std::vector<std::string> msgs( 1, it.tname() );
686 std::vector<item *> opts( 1, &it );
687
688 for( auto e : it.gunmods() ) {
689 if( e->is_gun() && !e->has_flag( "NO_UNLOAD" ) &&
690 ( e->magazine_current() || e->ammo_remaining() > 0 || e->casings_count() > 0 ) ) {
691 msgs.emplace_back( e->tname() );
692 opts.emplace_back( e );
693 }
694 }
695
696 item *target = nullptr;
697 if( opts.size() > 1 ) {
698 const int ret = uilist( _( "Unload what?" ), msgs );
699 if( ret >= 0 ) {
700 target = opts[ret];
701 }
702 } else {
703 target = &it;
704 }
705
706 if( target == nullptr ) {
707 return false;
708 }
709
710 // Next check for any reasons why the item cannot be unloaded
711 if( target->ammo_types().empty() || target->ammo_capacity() <= 0 ) {
712 add_msg( m_info, _( "You can't unload a %s!" ), target->tname() );
713 return false;
714 }
715
716 if( target->has_flag( "NO_UNLOAD" ) ) {
717 if( target->has_flag( "RECHARGE" ) || target->has_flag( "USE_UPS" ) ) {
718 add_msg( m_info, _( "You can't unload a rechargeable %s!" ), target->tname() );
719 } else {
720 add_msg( m_info, _( "You can't unload a %s!" ), target->tname() );
721 }
722 return false;
723 }
724
725 if( !target->magazine_current() && target->ammo_remaining() <= 0 && target->casings_count() <= 0 ) {
726 if( target->is_tool() ) {
727 add_msg( m_info, _( "Your %s isn't charged." ), target->tname() );
728 } else {
729 add_msg( m_info, _( "Your %s isn't loaded." ), target->tname() );
730 }
731 return false;
732 }
733
734 target->casings_handle( [&]( item & e ) {
735 return you.i_add_or_drop( e );
736 } );
737
738 if( target->is_magazine() ) {
739 // Calculate the time to remove the contained ammo (consuming half as much time as required to load the magazine)
740 int mv = 0;
741 int qty = 0;
742 std::vector<item *> remove_contained;
743 for( item *contained : it.contents.all_items_top() ) {
744 mv += you.item_reload_cost( it, *contained, contained->charges ) / 2;
745 if( add_or_drop_with_msg( you, *contained, true ) ) {
746 qty += contained->charges;
747 remove_contained.push_back( contained );
748 }
749 }
750 // remove the ammo leads in the belt
751 for( item *remove : remove_contained ) {
752 it.remove_item( *remove );
753 }
754
755 // remove the belt linkage
756 if( it.is_ammo_belt() ) {
757 if( it.type->magazine->linkage ) {
758 item link( *it.type->magazine->linkage, calendar::turn, qty );
759 add_or_drop_with_msg( you, link, true );
760 }
761 add_msg( _( "You disassemble your %s." ), it.tname() );
762 } else {
763 add_msg( _( "You unload your %s." ), it.tname() );
764 }
765
766 you.mod_moves( -std::min( 200, mv ) );
767 if( loc->has_flag( "MAG_DESTROY" ) && loc->ammo_remaining() == 0 ) {
768 loc.remove_item();
769 }
770 return true;
771
772 } else if( target->magazine_current() ) {
773 if( !add_or_drop_with_msg( you, *target->magazine_current(), true ) ) {
774 return false;
775 }
776 // Eject magazine consuming half as much time as required to insert it
777 you.moves -= you.item_reload_cost( *target, *target->magazine_current(), -1 ) / 2;
778
779 target->remove_items_with( [&target]( const item & e ) {
780 return target->magazine_current() == &e;
781 } );
782
783 } else if( target->ammo_remaining() ) {
784 int qty = target->ammo_remaining();
785
786 if( target->ammo_current() == itype_plut_cell ) {
787 qty = target->ammo_remaining() / PLUTONIUM_CHARGES;
788 if( qty > 0 ) {
789 add_msg( _( "You recover %i unused plutonium." ), qty );
790 } else {
791 add_msg( m_info, _( "You can't remove partially depleted plutonium!" ) );
792 return false;
793 }
794 }
795
796 // Construct a new ammo item and try to drop it
797 item ammo( target->ammo_current(), calendar::turn, qty );
798 if( target->is_filthy() ) {
799 ammo.set_flag( "FILTHY" );
800 }
801
802 if( ammo.made_of( LIQUID ) ) {
803 if( !add_or_drop_with_msg( you, ammo, false ) ) {
804 qty -= ammo.charges; // only handled part (or none) of the liquid
805 }
806 if( qty <= 0 ) {
807 return false; // no liquid was moved
808 }
809
810 } else if( !add_or_drop_with_msg( you, ammo, qty > 1 ) ) {
811 return false;
812 }
813
814 // If successful remove appropriate qty of ammo consuming half as much time as required to load it
815 you.moves -= you.item_reload_cost( *target, ammo, qty ) / 2;
816
817 if( target->ammo_current() == itype_plut_cell ) {
818 qty *= PLUTONIUM_CHARGES;
819 }
820
821 target->ammo_set( target->ammo_current(), target->ammo_remaining() - qty );
822 }
823
824 // Turn off any active tools
825 if( target->is_tool() && target->active && target->ammo_remaining() == 0 ) {
826 target->type->invoke( you, *target, you.pos() );
827 }
828
829 add_msg( _( "You unload your %s." ), target->tname() );
830 return true;
831
832}
