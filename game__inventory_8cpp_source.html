158static item_location inv_internal( player &u, const inventory_selector_preset &preset,
159 const std::string &title, int radius,
160 const std::string &none_message,
161 const std::string &hint = std::string() )
162{
163 inventory_pick_selector inv_s( u, preset );
164
165 inv_s.set_title( title );
166 inv_s.set_hint( hint );
167 inv_s.set_display_stats( false );
168
169 std::pair<size_t, size_t> init_pair;
170 bool init_selection = false;
171 std::string init_filter;
172 bool has_init_filter = false;
173
174 const std::vector<activity_id> consuming {
175 ACT_EAT_MENU,
176 ACT_CONSUME_FOOD_MENU,
177 ACT_CONSUME_DRINK_MENU,
178 ACT_CONSUME_MEDS_MENU };
179
180 if( u.has_activity( consuming ) && u.activity.values.size() >= 2 ) {
181 init_pair.first = u.activity.values[0];
182 init_pair.second = u.activity.values[1];
183 init_selection = true;
184 }
185 if( u.has_activity( consuming ) && !u.activity.str_values.empty() ) {
186 init_filter = u.activity.str_values[0];
187 has_init_filter = true;
188 }
189
190 do {
191 u.inv.restack( u );
192
193 inv_s.clear_items();
194 inv_s.add_character_items( u );
195 inv_s.add_nearby_items( radius );
196
197 if( has_init_filter ) {
198 inv_s.set_filter( init_filter );
199 has_init_filter = false;
200 }
201 // Set position after filter to keep cursor at the right position
202 if( init_selection ) {
203 inv_s.select_position( init_pair );
204 init_selection = false;
205 }
206
207 if( inv_s.empty() ) {
208 const std::string msg = none_message.empty()
209 ? _( "You don't have the necessary item at hand." )
210 : none_message;
211 popup( msg, PF_GET_KEY );
212 return item_location();
213 }
214
215 item_location location = inv_s.execute();
216
217 if( inv_s.keep_open ) {
218 inv_s.keep_open = false;
219 continue;
220 }
221
222 if( u.has_activity( consuming ) ) {
223 u.activity.values.clear();
224 init_pair = inv_s.get_selection_position();
225 u.activity.values.push_back( init_pair.first );
226 u.activity.values.push_back( init_pair.second );
227 u.activity.str_values.clear();
228 u.activity.str_values.emplace_back( inv_s.get_filter() );
229 }
230
231 return location;
232
233 } while( true );

644 bool sort_compare( const inventory_entry &lhs, const inventory_entry &rhs ) const override {
645 time_duration time_a = get_time_left( lhs.any_item() );
646 time_duration time_b = get_time_left( rhs.any_item() );
647 int order_a = get_order( lhs.any_item(), time_a );
648 int order_b = get_order( rhs.any_item(), time_b );
649
650 return order_a < order_b
651 || ( order_a == order_b && time_a < time_b )
652 || ( order_a == order_b && time_a == time_b &&
653 inventory_selector_preset::sort_compare( lhs, rhs ) );
654 }
