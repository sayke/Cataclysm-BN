197

202 JsonOut jsout( stream );
203 jsout.start_object();
204 for( const mod_id &mod : mods ) {
205 jsout.member( mod.str() );

213bool load_world_lua_state( const std::string &path )
214{
215 lua_state &state = *DynamicDataLoader::get_instance().lua;
216 const mod_management::t_mod_list &mods = world_generator->active_world->active_mod_order;
217 sol::table t = get_mod_storage_table( state );
218
219 bool ret = read_from_file_optional( path, [&]( std::istream & stream ) {
220 JsonIn jsin( stream );
221 JsonObject jsobj = jsin.get_object();
222
223 for( const mod_id &mod : mods ) {
224 if( !jsobj.has_object( mod.str() ) ) {
225 // Mod could have been added to existing save
226 continue;
227 }
228 JsonObject mod_obj = jsobj.get_object( mod.str() );
229 deserialize_lua_table( t[mod.str()], mod_obj );
230 }
231 } );
232
233 run_on_game_load_hooks( state );
234 return ret;

252void init_global_state_tables( lua_state &state, const std::vector<mod_id> &modlist )
253{
254 sol::state &lua = state.lua;
255
256 sol::table active_mods = lua.create_table();
257 sol::table mod_runtime = lua.create_table();
258 sol::table mod_storage = lua.create_table();
259 sol::table hooks = lua.create_table();
260
261 for( size_t i = 0; i < modlist.size(); i++ ) {
262 active_mods[ i + 1 ] = modlist[i].str();
263 mod_runtime[ modlist[i].str() ] = lua.create_table();
264 mod_storage[ modlist[i].str() ] = lua.create_table();
265 }
266
267 // Main game data table
268 sol::table gt = lua.globals()["game"];
269
270 // Internal table that bypasses read-only facades
271 sol::table it = lua.create_table();
272 gt["cata_internal"] = it;
273 it["active_mods"] = active_mods;
274 it["mod_runtime"] = mod_runtime;
275 it["mod_storage"] = mod_storage;
276 it["on_every_x_hooks"] = std::vector<cata::on_every_x_hooks>();
277 gt["hooks"] = hooks;
278
279 // Runtime infrastructure
280 gt["active_mods"] = make_readonly_table( lua, active_mods );
281 gt["mod_runtime"] = make_readonly_table( lua, mod_runtime );
282 gt["mod_storage"] = make_readonly_table( lua, mod_storage );
283 gt["hooks"] = make_readonly_table( lua, hooks );
284
285 // iuse functions
286 gt["iuse_functions"] = lua.create_table();
287
288 // hooks
289 hooks["on_game_load"] = lua.create_table();
290 hooks["on_game_save"] = lua.create_table();
291 hooks["on_mapgen_postprocess"] = lua.create_table();

340void run_hooks( lua_state &state, std::string_view hooks_table, Args &&...args )
341{
342 sol::state &lua = state.lua;
343 sol::table hooks = lua.globals()["game"]["hooks"][hooks_table];
344 for( auto &ref : hooks ) {
345 int idx = -1;
346 try {
347 idx = ref.first.as<int>();
348 sol::protected_function func = ref.second;
349 sol::protected_function_result res = func( std::forward<Args>( args )... );
350 check_func_result( res );
351 } catch( std::runtime_error &e ) {
352 debugmsg( "Failed to run hook %s[%d]: %s", hooks_table, idx, e.what() );
353 break;
354 }
355 }
