147{
148 item &it = i_at( position );
149 if( it.is_null() ) {
150 debugmsg( "invalid item position %d for reduce_charges", position );
151 return item();
152 }

200bool character_martial_arts::pick_style( const avatar &you ) // Style selection menu
201{
202 enum style_selection {
203 KEEP_HANDS_FREE = 0,
204 STYLE_OFFSET
205 };
206
207 // If there are style already, cursor starts there
208 // if no selected styles, cursor starts from no-style
209
210 // Any other keys quit the menu
211 const std::vector<matype_id> &selectable_styles = you.has_active_bionic(
212 bio_cqb ) ? bio_cqb_styles :
213 ma_styles;
214
215 input_context ctxt( "MELEE_STYLE_PICKER" );
216 ctxt.register_action( "SHOW_DESCRIPTION" );
217
218 uilist kmenu;
219 kmenu.text = string_format( _( "Select a style.\n"
220 "\n"
221 "STR: <color_white>%d</color>, DEX: <color_white>%d</color>, "
222 "PER: <color_white>%d</color>, INT: <color_white>%d</color>\n"
223 "Press [<color_yellow>%s</color>] for more info.\n" ),
224 you.get_str(), you.get_dex(), you.get_per(), you.get_int(),
225 ctxt.get_desc( "SHOW_DESCRIPTION" ) );
226 ma_style_callback callback( static_cast<size_t>( STYLE_OFFSET ), selectable_styles );
227 kmenu.callback = &callback;
228 kmenu.input_category = "MELEE_STYLE_PICKER";
229 kmenu.additional_actions.emplace_back( "SHOW_DESCRIPTION", translation() );
230 kmenu.desc_enabled = true;
231 kmenu.addentry_desc( KEEP_HANDS_FREE, true, 'h',
232 keep_hands_free ? _( "Keep hands free (on)" ) : _( "Keep hands free (off)" ),
233 _( "When this is enabled, player won't wield things unless explicitly told to." ) );
234
235 kmenu.selected = STYLE_OFFSET;
236
237 for( size_t i = 0; i < selectable_styles.size(); i++ ) {
238 auto &style = selectable_styles[i].obj();
239 //Check if this style is currently selected
240 const bool selected = selectable_styles[i] == style_selected;
241 std::string entry_text = style.name.translated();
242 if( selected ) {
243 kmenu.selected = i + STYLE_OFFSET;
244 entry_text = colorize( entry_text, c_pink );
245 }
246 kmenu.addentry_desc( i + STYLE_OFFSET, true, -1, entry_text, style.description.translated() );
247 }
248
249 kmenu.query();
250 int selection = kmenu.ret;
251
252 if( selection >= STYLE_OFFSET ) {
253 style_selected = selectable_styles[selection - STYLE_OFFSET];
254 martialart_use_message( you );
255 } else if( selection == KEEP_HANDS_FREE ) {
256 keep_hands_free = !keep_hands_free;
257 } else {
258 return false;
259 }
260
261 return true;

277recipe_subset player::get_recipes_from_books( const inventory &crafting_inv,
278 recipe_filter filter ) const
279{
280 recipe_subset res;
281
282 for( const auto &stack : crafting_inv.const_slice() ) {
283 const item &candidate = stack->front();
284
285 for( std::pair<const recipe *, int> recipe_entry :
286 candidate.get_available_recipes( *this ) ) {
287 if( filter && !filter( *recipe_entry.first ) ) {
288 continue;
289 }
290 res.include( recipe_entry.first, recipe_entry.second );
291 }
292 }
293
294 return res;
295}

297recipe_subset player::get_available_recipes( const inventory &crafting_inv,
298 const std::vector<npc *> *helpers, recipe_filter filter ) const
299{
300 recipe_subset res;
301
302 if( filter ) {
303 res.include_if( get_learned_recipes(), filter );
304 } else {
305 res.include( get_learned_recipes() );
306 }
307
308 res.include( get_recipes_from_books( crafting_inv, filter ) );
309
310 if( helpers != nullptr ) {
311 for( npc *np : *helpers ) {
312 // Directly form the helper's inventory
313 res.include( get_recipes_from_books( np->inv, filter ) );
314 // Being told what to do
315 res.include_if( np->get_learned_recipes(), [ this, &filter ]( const recipe & r ) {
316 if( filter && !filter( r ) ) {
317 return false;
318 }
319 // Skilled enough to understand
320 return get_skill_level( r.skill_used ) >= static_cast<int>( r.difficulty * 0.8f );
321 } );
322 }
323 }
324
325 return res;
326}
