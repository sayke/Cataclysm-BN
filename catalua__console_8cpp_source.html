26static std::vector<folded_log_msg> build_folded_log( int width )
27{
28 std::vector<folded_log_msg> ret;
29 for( const lua_log_msg &msg : get_lua_log_instance().get_entries() ) {
30 std::vector<std::string> lines = foldstring( msg.text, width );
31 for( int i = static_cast<int>( lines.size() ) - 1; i >= 0; i-- ) {
32 ret.push_back( folded_log_msg{ i == 0, msg.level, std::move( lines[i] ) } );
33 }
34 }
35 return ret;
36}

67static void add_to_input_history( const std::string &s )
68{
69 std::vector<std::string> &hist = get_input_history();
70 for( auto it = hist.begin(); it != hist.end(); it++ ) {
71 if( *it == s ) {
72 // Refresh existing history entry
73 std::string msg = std::move( *it );
74 hist.erase( it );
75 hist.push_back( std::move( msg ) );
76 return;
77 }
78 }
79 // Add new history entry
80 hist.push_back( s );
