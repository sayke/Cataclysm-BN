169void deserialize_lua_table( sol::table t, JsonObject &obj )
170{
171 sol::state_view lua( t.lua_state() );
172
173 for( JsonMember it : obj ) {
174 std::string key = it.name();
175 if( it.test_bool() ) {
176 t[key] = it.get_bool();
177 } else if( it.test_string() ) {
178 t[key] = it.get_string();
179 } else if( it.test_object() ) {
180 JsonObject jo = it.get_object();
181 std::string entry_type = jo.get_member( "type" );
182 if( entry_type == "int" ) {
183 int data = jo.get_int( "data" );
184 t[key] = data;
185 } else if( entry_type == "float" ) {
186 double data = jo.get_float( "data" );
187 t[key] = data;
188 } else if( entry_type == "lua_table" ) {
189 JsonObject data = jo.get_object( "data" );
190 sol::table new_table = lua.create_table();
191 deserialize_lua_table( new_table, data );
192 t[key] = new_table;
193 } else if( entry_type == "userdata" ) {
194 std::string kind = jo.get_member( "kind" );
195 JsonIn &data_raw = *jo.get_raw( "data" );
196 // Horrible hack ahead
197 std::string script = string_format( "return %s.new()", kind );
198 sol::protected_function_result res = lua.script( script, "deserialize_init", sol::load_mode::any );
199 if( res.status() != sol::call_status::ok ) {
200 debugmsg( "Failed to init container for deserializable type '%s'", kind );
201 } else {
202 sol::object obj = res;
203 sol::table obj_table = obj.as<sol::table>();
204 if( !obj_table ) {
205 debugmsg( "Failed to init container for deserializable type '%s'", kind );
206 } else {
207 sol::protected_function deserialize_func = obj_table["deserialize"];
208 if( !deserialize_func ) {
209 debugmsg( "Failed to deserialize type '%s': not deserializable.", kind );
210 } else {
211 sol::protected_function_result res = deserialize_func( obj, data_raw );
212 if( res.status() != sol::call_status::ok ) {
213 sol::error err = res;
214 debugmsg( "Failed to deserialize type '%s': %s", kind, err.what() );
215 } else {
216 t[key] = obj;
217 }
218 }
219 }
220 }
221 } else {
222 debugmsg( "Deserialization of record type '%s' is not implemented.", entry_type );
223 }
224 } else if( it.test_array() ) {
225 debugmsg( "Arrays are not supported when deserializing Lua table." );
