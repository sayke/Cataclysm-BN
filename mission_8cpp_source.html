132void mission::on_creature_death( Creature &poor_dead_dude )
133{
134 if( poor_dead_dude.is_hallucination() ) {
135 return;
136 }
137 monster *mon = dynamic_cast<monster *>( &poor_dead_dude );
138 if( mon != nullptr ) {
139 if( mon->mission_id == -1 ) {
140 return;
141 }
142 const auto mission = mission::find( mon->mission_id );
143 const auto type = mission->type;
144 if( type->goal == MGOAL_FIND_MONSTER ) {
145 mission->fail();
146 }
147 if( type->goal == MGOAL_KILL_MONSTER ) {
148 mission->step_complete( 1 );
149 }
150 return;
151 }
152 npc *p = dynamic_cast<npc *>( &poor_dead_dude );
153 if( p == nullptr ) {
154 // Must be the player
155 for( auto &miss : g->u.get_active_missions() ) {
156 // mission is free and can be reused
157 miss->player_id = character_id();
158 }
159 // The missions remains assigned to the (dead) character. This should not cause any problems
160 // as the character is dismissed anyway.
161 // Technically, the active missions could be moved to the failed mission section.
162 return;
163 }
164 const auto dead_guys_id = p->getID();
165 for( auto &e : world_missions ) {
166 mission &i = e.second;
167 if( !i.in_progress() ) {
168 continue;
169 }
170 //complete the mission if you needed killing
171 if( i.type->goal == MGOAL_ASSASSINATE && i.target_npc_id == dead_guys_id ) {
172 i.step_complete( 1 );
173 }
174 //fail the mission if the mission giver dies
175 if( i.npc_id == dead_guys_id ) {
176 i.fail();
177 }
178 //fail the mission if recruit target dies
179 if( i.type->goal == MGOAL_RECRUIT_NPC && i.target_npc_id == dead_guys_id ) {
180 i.fail();
181 }
182 }

280void mission::wrap_up()
281{
282 auto &u = g->u;
283 if( u.getID() != player_id ) {
284 // This is called from npctalk.cpp, the npc should only offer the option to wrap up mission
285 // that have been assigned to the current player.
286 debugmsg( "mission::wrap_up called, player %d was assigned, but current player is %d",
287 player_id.get_value(), u.getID().get_value() );
288 }
289
290 status = mission_status::success;
291 u.on_mission_finished( *this );
292 std::vector<item_comp> comps;
293 switch( type->goal ) {
294 case MGOAL_FIND_ITEM_GROUP: {
295 inventory tmp_inv = u.crafting_inventory();
296 std::vector<item *> items = std::vector<item *>();
297 tmp_inv.dump( items );
298 item_group_id grp_type = type->group_id;
299 itype_id container = type->container_id;
300 bool specific_container_required = !container.is_null();
301 bool remove_container = type->remove_container;
302 itype_id empty_container = type->empty_container;
303
304 std::map<itype_id, int> matches = std::map<itype_id, int>();
305 get_all_item_group_matches(
306 items, grp_type, matches,
307 container, itype_id::NULL_ID(), specific_container_required );
308
309 for( std::pair<const itype_id, int> &cnt : matches ) {
310 if( cnt.second >= type->item_count ) {
311 comps.push_back( item_comp( cnt.first, type->item_count ) );
312 }
313 }
314
315 u.consume_items( comps );
316
317 if( remove_container ) {
318 std::vector<item_comp> container_comp = std::vector<item_comp>();
319 if( !empty_container.is_null() ) {
320 container_comp.push_back( item_comp( empty_container, type->item_count ) );
321 u.consume_items( container_comp );
322 } else {
323 container_comp.push_back( item_comp( container, type->item_count ) );
324 u.consume_items( container_comp );
325 }
326 }
327 }
328 break;
329
330 case MGOAL_FIND_ITEM:
331 comps.push_back( item_comp( type->item_id, item_count ) );
332 u.consume_items( comps );
333 break;
334 case MGOAL_FIND_ANY_ITEM:
335 u.remove_mission_items( uid );
336 break;
337 default:
338 //Suppress warnings
339 break;
340 }
341
342 type->end( this );

345bool mission::is_complete( const character_id &_npc_id ) const
346{
347 if( status == mission_status::success ) {
348 return true;
349 }
350
351 auto &u = g->u;
352 switch( type->goal ) {
353 case MGOAL_GO_TO: {
354 const tripoint_abs_omt cur_pos = u.global_omt_location();
355 return ( rl_dist( cur_pos, target ) <= 1 );
356 }
357
358 case MGOAL_GO_TO_TYPE: {
359 const auto cur_ter = overmap_buffer.ter( g->u.global_omt_location() );
360 return is_ot_match( type->target_id.str(), cur_ter, ot_match_type::type );
361 }
362
363 case MGOAL_FIND_ITEM_GROUP: {
364 inventory tmp_inv = u.crafting_inventory();
365 std::vector<item *> items = std::vector<item *>();
366 tmp_inv.dump( items );
367 item_group_id grp_type = type->group_id;
368 itype_id container = type->container_id;
369 bool specific_container_required = !container.is_null();
370
371 std::map<itype_id, int> matches = std::map<itype_id, int>();
372 get_all_item_group_matches(
373 items, grp_type, matches,
374 container, itype_id::NULL_ID(), specific_container_required );
375
376 for( std::pair<const itype_id, int> &pair : matches ) {
377 if( pair.second >= type->item_count ) {
378 return true;
379 }
380 }
381 }
382 return false;
383
384 case MGOAL_FIND_ITEM: {
385 if( npc_id.is_valid() && npc_id != _npc_id ) {
386 return false;
387 }
388 const inventory &tmp_inv = u.crafting_inventory();
389 // TODO: check for count_by_charges and use appropriate player::has_* function
390 if( !tmp_inv.has_amount( type->item_id, item_count ) ) {
391 return tmp_inv.has_amount( type->item_id, 1 ) && tmp_inv.has_charges( type->item_id, item_count );
392 }
393 }
394 return true;
395
396 case MGOAL_FIND_ANY_ITEM:
397 return u.has_mission_item( uid ) && ( !npc_id.is_valid() || npc_id == _npc_id );
398
399 case MGOAL_FIND_MONSTER:
400 if( npc_id.is_valid() && npc_id != _npc_id ) {
401 return false;
402 }
403 return g->get_creature_if( [&]( const Creature & critter ) {
404 const monster *const mon_ptr = dynamic_cast<const monster *>( &critter );
405 return mon_ptr && mon_ptr->mission_id == uid;
406 } );
407
408 case MGOAL_RECRUIT_NPC: {
409 npc *p = g->find_npc( target_npc_id );
410 return p != nullptr && p->is_player_ally();
411 }
412
413 case MGOAL_RECRUIT_NPC_CLASS: {
414 const auto npcs = overmap_buffer.get_npcs_near_player( 100 );
415 for( auto &npc : npcs ) {
416 if( npc->myclass == recruit_class && npc->is_player_ally() ) {
417 return true;
418 }
419 }
420 return false;
421 }
422
423 case MGOAL_FIND_NPC:
424 return npc_id == _npc_id;
425
426 case MGOAL_TALK_TO_NPC:
427 case MGOAL_ASSASSINATE:
428 case MGOAL_KILL_MONSTER:
429 case MGOAL_COMPUTER_TOGGLE:
430 return step >= 1;
431
432 case MGOAL_KILL_MONSTER_TYPE:
433 return g->get_kill_tracker().kill_count( monster_type ) >= kill_count_to_reach;
434
435 case MGOAL_KILL_MONSTER_SPEC:
436 return g->get_kill_tracker().kill_count( monster_species ) >= kill_count_to_reach;
437
438 case MGOAL_CONDITION: {
439 // For now, we only allow completing when talking to the mission originator.
440 if( npc_id != _npc_id ) {
441 return false;
442 }
443
444 npc *n = g->find_npc( _npc_id );
445 if( n == nullptr ) {
446 return false;
447 }
448
449 mission_goal_condition_context cc;
450 cc.alpha = &u;
451 cc.beta = n;
452
453 for( auto &mission : n->chatbin.missions_assigned ) {
454 if( mission->get_assigned_player_id() == g->u.getID() ) {
455 cc.missions_assigned.push_back( mission );
456 }
457 }
458
459 return type->test_goal_condition( cc );
460 }
461
462 default:
463 return false;
464 }

467void mission::get_all_item_group_matches( std::vector<item *> &items,
468 item_group_id &grp_type, std::map<itype_id, int> &matches,
469 const itype_id &required_container, const itype_id &actual_container,
470 bool &specific_container_required )
471{
472 for( item *itm : items ) {
473 bool correct_container = ( required_container == actual_container ) ||
474 !specific_container_required;
475
476 bool item_in_group = item_group::group_contains_item( grp_type, itm->typeId() );
477
478 //check whether item itself is target
479 if( item_in_group && correct_container ) {
480 matches[ itm->typeId() ] += itm->count();
481 }
482
483 //recursivly check item contents for target
484 if( itm->is_container() && !itm->is_container_empty() ) {
485 std::list<item *> content_list = itm->contents.all_items_top();
486 std::vector<item *> content = std::vector<item *>();
487
488 //list of item into list item*
489 std::transform(
490 content_list.begin(), content_list.end(),
491 std::back_inserter( content ),
492 []( item * p ) {
493 return p;
494 } );
495
496 get_all_item_group_matches(
497 content, grp_type, matches,
498 required_container, ( itm->typeId() ), specific_container_required );
499 }
500 }
