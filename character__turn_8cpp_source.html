321 }

808void Character::process_items()
809{
810 item &weapon = primary_weapon();
811 if( weapon.needs_processing() && weapon.process( as_player(), pos(), false ) ) {
812 weapon = item();
813 }
814
815 std::vector<item *> inv_active = inv.active_items();
816 for( item *tmp_it : inv_active ) {
817 if( tmp_it->process( as_player(), pos(), false ) ) {
818 inv.remove_item( tmp_it );

821
822 // worn items
823 remove_worn_items_with( [this]( item & itm ) {
824 return itm.needs_processing() && itm.process( as_player(), pos(), false );
825 } );
826
827 // Active item processing done, now we're recharging.
828 std::vector<item *> active_worn_items;
829 bool weapon_active = weapon.has_flag( "USE_UPS" ) &&
830 weapon.charges < weapon.type->maximum_charges();
831 std::vector<size_t> active_held_items;
832 int ch_UPS = 0;
833 for( size_t index = 0; index < inv.size(); index++ ) {
834 item &it = inv.find_item( index );
835 itype_id identifier = it.type->get_id();
836 if( identifier == itype_UPS_off ) {
837 ch_UPS += it.ammo_remaining();
838 } else if( identifier == itype_adv_UPS_off ) {
839 ch_UPS += it.ammo_remaining() / 0.6;
840 }
841 if( it.has_flag( "USE_UPS" ) && it.charges < it.type->maximum_charges() ) {
842 active_held_items.push_back( index );
843 }
844 }
845 bool update_required = get_check_encumbrance();
846 for( item &w : worn ) {
847 if( w.has_flag( "USE_UPS" ) &&
848 w.charges < w.type->maximum_charges() ) {
849 active_worn_items.push_back( &w );
850 }
851 // Necessary for UPS in Aftershock - check worn items for charge
852 const itype_id &identifier = w.typeId();
853 if( identifier == itype_UPS_off ) {
854 ch_UPS += w.ammo_remaining();
855 } else if( identifier == itype_adv_UPS_off ) {
856 ch_UPS += w.ammo_remaining() / 0.6;
857 }
858 if( !update_required && w.encumbrance_update_ ) {
859 update_required = true;
860 }
861 w.encumbrance_update_ = false;
862 }
863 if( update_required ) {
864 reset_encumbrance();
865 }
866 if( has_active_bionic( bionic_id( "bio_ups" ) ) ) {
867 ch_UPS += units::to_kilojoule( get_power_level() );
868 }
869 int ch_UPS_used = 0;
870
871 // Load all items that use the UPS to their minimal functional charge,
872 // The tool is not really useful if its charges are below charges_to_use
873 for( size_t index : active_held_items ) {
874 if( ch_UPS_used >= ch_UPS ) {
875 break;
876 }
877 item &it = inv.find_item( index );
878 ch_UPS_used++;
879 it.charges++;
880 }
881 if( weapon_active && ch_UPS_used < ch_UPS ) {
882 ch_UPS_used++;
883 weapon.charges++;
884 }
885 for( item *worn_item : active_worn_items ) {
886 if( ch_UPS_used >= ch_UPS ) {
887 break;
888 }
889 ch_UPS_used++;
890 worn_item->charges++;
891 }
892 if( ch_UPS_used > 0 ) {
893 use_charges( itype_UPS, ch_UPS_used );
894 }

900void update_body_wetness( Character &who, const w_point &weather )
901{
902 // Average number of turns to go from completely soaked to fully dry
903 // assuming average temperature and humidity
904 constexpr time_duration average_drying = 2_hours;
905
906 // A modifier on drying time
907 double delay = 1.0;
908 // Weather slows down drying
909 delay += ( ( weather.humidity - 66 ) - ( units::to_fahrenheit( weather.temperature ) - 65 ) ) / 100;
910 delay = std::max( 0.1, delay );
911 // Fur/slime retains moisture
912 if( who.has_trait( trait_LIGHTFUR ) ||
913 who.has_trait( trait_FUR ) ||
914 who.has_trait( trait_FELINE_FUR ) ||
915 who.has_trait( trait_LUPINE_FUR ) ||
916 who.has_trait( trait_CHITIN_FUR ) ||
917 who.has_trait( trait_CHITIN_FUR2 ) ||
918 who.has_trait( trait_CHITIN_FUR3 ) ) {
919 delay = delay * 6 / 5;
920 }
921 if( who.has_trait( trait_URSINE_FUR ) || who.has_trait( trait_SLIMY ) ) {
922 delay *= 1.5;
923 }
924
925 if( !x_in_y( 1, to_turns<int>( average_drying * delay / 100.0 ) ) ) {
926 // No drying this turn
927 return;
928 }
929
930 // Now per-body-part stuff
931 // To make drying uniform, make just one roll and reuse it
932 const int drying_roll = rng( 1, 80 );
933
934 for( const body_part bp : all_body_parts ) {
935 if( who.body_wetness[bp] == 0 ) {
936 continue;
937 }
938 // This is to normalize drying times
939 int drying_chance = who.drench_capacity[bp];
940 // Body temperature affects duration of wetness
941 // Note: Using temp_conv rather than temp_cur, to better approximate environment
942 if( who.temp_conv[bp] >= BODYTEMP_SCORCHING ) {
943 drying_chance *= 2;
944 } else if( who.temp_conv[bp] >= BODYTEMP_VERY_HOT ) {
945 drying_chance = drying_chance * 3 / 2;
946 } else if( who.temp_conv[bp] >= BODYTEMP_HOT ) {
947 drying_chance = drying_chance * 4 / 3;
948 } else if( who.temp_conv[bp] > BODYTEMP_COLD ) {
949 // Comfortable, doesn't need any changes
950 } else {
951 // Evaporation doesn't change that much at lower temp
952 drying_chance = drying_chance * 3 / 4;
953 }
954
955 if( drying_chance < 1 ) {
956 drying_chance = 1;
957 }
958
959 // TODO: Make evaporation reduce body heat
960 if( drying_chance >= drying_roll ) {
961 who.body_wetness[bp] -= 1;
962 if( who.body_wetness[bp] < 0 ) {
963 who.body_wetness[bp] = 0;
964 }
965 }
966 }
