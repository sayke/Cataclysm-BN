5/**
6 * Due to some strange behavior in sol2, we need to define
7 * comparison operators for types that are referenced by string_id or int_id.
8 * For details, see https://github.com/ThePhD/sol2/issues/1264
9 * string_id and int_id essentially act like smart pointers
10 * and implement comparison operators, meaning T also has to implement them.
11 *
12 * You can use this macro as a quick solution, just pass the type as T
13 * and the member/method that contains/returns the id as id_getter.
14 *
15 * Example implementation:
16 * ```c++
17 * class ter_t {
18 * ter_str_id id;
19 * ...
20 * LUA_TYPE_OPS( ter_t, id );
21 * ...
22 * }
23 * ```
24 */
25#define LUA_TYPE_OPS( T, id_getter ) \
26 inline bool operator==( const T &rhs ) const { \
27 return id_getter == rhs.id_getter; \
28 }; \
29 inline bool operator<( const T &rhs ) const { \
30 return id_getter < rhs.id_getter; \
31 }

33#endif // CATA_SRC_CATALUA_TYPE_OPERATORS_H
