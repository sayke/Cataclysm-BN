15/**
16 * Adaptor between UI code and the UI management system. Using this class allows
17 * UIs to be correctly resized and redrawn when the game window is resized or
18 * when exiting from other UIs.
19 *
20 * Usage example:
21 * ```
22 * // Effective in the local scope
23 * ui_adaptor ui;
24 * // Ncurses window for drawing
25 * catacurses::window win;
26 * // Things to do when the game window changes size
27 * ui.on_screen_resize( [&]( ui_adaptor & ui ) {
28 * // Create an ncurses window
29 * win = catacurses::newwin( TERMX / 2, TERMY / 2, point( TERMX / 4, TERMY / 4 ) );
30 * // The window passed to this call must contain all the space the redraw
31 * // callback draws to, to ensure proper refreshing when resizing or exiting
32 * // from other UIs.
33 * ui.position_from_window( win );
34 * } );
35 * // Mark the resize callback to be called on the first redraw
36 * ui.mark_resize();
37 * // Things to do when redrawing the UI
38 * ui.on_redraw( [&]( const ui_adaptor & ) {
39 * // Clear UI area
40 * werase( win );
41 * // Print things
42 * mvwprintw( win, point_zero, "Hello World!" );
43 * // Write to frame buffer
44 * wnoutrefresh( win );
45 * } );
46 *
47 * input_context ctxt( "<CATEGORY_NAME>" );
48 * ctxt.register_action( "QUIT" );
49 * while( true ) {
50 * // Invalidate the top UI (that is, this UI) and redraw all
51 * // invalidated UIs (including lower UIs that calls this UI).
52 * // May call the resize callbacks.
53 * ui_manager::redraw();
54 * // Get user input. Note that this may call the resize and redraw
55 * // callbacks multiple times due to screen resize, rendering target
56 * // reset, etc.
57 * const std::string action = ctxt.handle_input();
58 * if( action == "QUIT" ) {
59 * break;
60 * }
61 * }
62 * ```
63 **/

119 /**
120 * Set redraw and resize callbacks. The resize callback should
121 * call `position` or `position_from_window` to set the size of the UI,
122 * and (re-)calculate any UI data that is related to the screen size,
123 * including `catacurses::window` instances. In most cases, you should
124 * also call `mark_resize` along with `on_screen_resize` so the UI is
125 * initialized by the resizing callback when redrawn for the first time.
126 *
127 * The redraw callback should only redraw to the area specified by the
128 * `position` or `position_from_window` call. Content drawn outside this
129 * area may not render correctly when resizing or exiting from other UIs.

131 *
132 * These callbacks should NOT:
133 * - Construct new `ui_adaptor` instances
134 * - Deconstruct old `ui_adaptor` instances
